50/**
51 * @brief Merge all the SVGs in the m_domList list into a single SVG
52 * @return The merged SVG as a DOM document
53 * @details m_domList contains DOM representations of multiple SVGs
54 Assuming each svg contains a main group containing every graphical item of the svg,
55 this function adds to the first svg of the list all the main groups of the other svgs,
56 thus creating a single svg with each svg of the list in it, every svg being in its own group.
57 This function is used in order to create svgs containing groups
58 */
59QDomDocument SvgGenerator::mergeSvgDoms()
60{
61 if (m_domList.isEmpty()) {
62 qDebug() << "Error : the SVG list is empty";
63 return QDomDocument();
64 }
65
66 QDomDocument mergedSvg = m_domList.at(0);
67
68 QDomElement mergedSvgRoot = mergedSvg.documentElement();
69 if (mergedSvgRoot.tagName() != "svg") {
70 qDebug() << "Error : the first SVG does not contain a <svg> tag.";
71 return QDomDocument();
72 }
73
74 for (int i = 1; i < m_domList.size(); ++i) {
75 QDomDocument domSvg = m_domList.at(i);
76 QDomElement svgRoot = domSvg.documentElement();
77 if (svgRoot.tagName() != "svg") {
78 qDebug() << "Error : the SVG does not contain a <svg> tag.";
79 return QDomDocument();
80 }
81 QDomNodeList svgGroups = svgRoot.elementsByTagName("g");
82 if (svgGroups.isEmpty()) {
83 qDebug() << "Error : the SVG does not contain a <g> tag.";
84 return QDomDocument();
85 }
86 QDomElement mainGroup = svgGroups.at(0).toElement();
87 mergedSvgRoot.appendChild(mainGroup);
88 }
89
90 return mergedSvg;
91}

95/**
96 * @brief Add a new SVG to the list of SVGs to be merged into a single SVG
97 * @param scene : the scene that must be converted to SVG
98 * @return void
99 * @details This function creates a SVG from the given scene and converts it into
100 a DOM that is added to the m_domList list of SVGs to be merged.
101*/
102void SvgGenerator::addSvgFromScene(QGraphicsScene *scene)
103{
104
105
106 QByteArray byteArray;
107 QBuffer buffer(&byteArray);
108 buffer.open(QIODevice::WriteOnly);
109
110 QSvgGenerator svgGenerator;
111 svgGenerator.setOutputDevice(&buffer);
112 svgGenerator.setSize(m_paper->rect().size().toSize());
113 svgGenerator.setViewBox(m_paper->rect());
114 svgGenerator.setTitle(QObject::tr("Pattern"));
115 svgGenerator.setDescription(m_description);
116 svgGenerator.setResolution(m_resolution);
117
118 QPainter painter;
119 painter.begin(&svgGenerator);
120 painter.setFont( QFont( "Arial", 8, QFont::Normal ) );
121 painter.setRenderHint(QPainter::Antialiasing, true);
122 painter.setBrush ( QBrush ( Qt::NoBrush ) );
123 scene->render(&painter, m_paper->rect(), m_paper->rect(), Qt::IgnoreAspectRatio);
124 painter.end();
125
126 QDomDocument domDoc;
127 if (domDoc.setContent(byteArray)) {
128 m_domList.append(domDoc);
129 } else {
130 qDebug() << "Error : Impossible to load the SVG content in the QDomDocument.";
131 }
132
133 buffer.close();
134}

138/**
139 * @brief Generate the merged SVG where each previously given scene is grouped separately.
140 * @return void
141 * @details This function merges the SVGs of the m_domList list and writes the result
142 in a file at the path given in the constructor.
143*/
144void SvgGenerator::generate()
145{
146
147
148 QDomDocument mergedSvg = mergeSvgDoms();
149
150 QFile outputFile(m_filepath);
151 if (!outputFile.open(QIODevice::WriteOnly | QIODevice::Text)) {
152 qDebug() << "Error : Couldn't write the output file.";
153 return;
154 }
155
156 QTextStream stream(&outputFile);
157 stream << mergedSvg.toString();
158 outputFile.close();
159
160 qDebug() << "Merged SVG Generated!";
161}
