127QString VAbstractConverter::getVersionStr() const
128{
129 const QDomNodeList nodeList = this->elementsByTagName(TagVersion);
130 if (nodeList.isEmpty())
131 {
132 const QString errorMsg(tr("Couldn't get version information."));
133 throw VException(errorMsg);
134 }
135
136 if (nodeList.count() > 1)
137 {
138 const QString errorMsg(tr("Too many tags <%1> in file.").arg(TagVersion));
139 throw VException(errorMsg);
140 }
141
142 const QDomNode domNode = nodeList.at(0);
143 if (domNode.isNull() == false && domNode.isElement())
144 {
145 const QDomElement domElement = domNode.toElement();
146 if (domElement.isNull() == false)
147 {
148 qInfo() << " VAbstractConverter::getVersionStr()" << domElement.text();
149 return domElement.text();
150 }
151 }
152 return QString(QStringLiteral("0.0.0"));
153}

155//---------------------------------------------------------------------------------------------------------------------
156int VAbstractConverter::GetVersion(const QString &version)
157{
158 ValidateVersion(version);
159
160 const QStringList ver = version.split(".");
161
162 bool ok = false;
163 const int major = ver.at(0).toInt(&ok);
164 if (!ok)
165 {
166 return 0x0;
167 }
168
169 ok = false;
170 const int minor = ver.at(1).toInt(&ok);
171 if (!ok)
172 {
173 return 0x0;
174 }
175
176 ok = false;
177 const int patch = ver.at(2).toInt(&ok);
178 if (!ok)
179 {
180 return 0x0;
181 }
182 return (major<<16)|(minor<<8)|(patch);
183}

185// @brief Removes single or repeated '.bak' extension (as long as it is at the end of \arg fileName)
186// @Param filename file to backup
187// @Return newfileName
188QString VAbstractConverter::removeBakExtension(const QString &fileName)
189{
190 QString newFileName = fileName;
191 int dotPos = newFileName.lastIndexOf(QLatin1Char('.'));
192
193 while (dotPos > -1 && newFileName.mid(dotPos) == QLatin1String(".bak"))
194 {
195 newFileName = newFileName.left(dotPos);
196 dotPos = newFileName.lastIndexOf(QLatin1Char('.'));
197 }
198
199 return newFileName;
200}

203void VAbstractConverter::ValidateVersion(const QString &version)
204{
205 const QRegularExpression rx(QStringLiteral("^(0|([1-9][0-9]*)).(0|([1-9][0-9]*)).(0|([1-9][0-9]*))$"));
206
207 if (rx.match(version).hasMatch() == false)
208 {
209 const QString errorMsg(tr("Version \"%1\" invalid.").arg(version));
210 throw VException(errorMsg);
211 }
212
213 if (version == QLatin1String("0.0.0"))
214 {
215 const QString errorMsg(tr("Version \"0.0.0\" invalid."));
216 throw VException(errorMsg);
217 }
218}

220//---------------------------------------------------------------------------------------------------------------------
221void VAbstractConverter::saveBackupFile() const
222{
223 //It's not possible in all cases to convert file without a loss of data.
224 //For such cases we will save a backup of the old version.
225 if (qApp->Settings()->getConvertBackupEnabled())
226 {
227 QString error;
228 const QFileInfo info(removeBakExtension(m_convertedFileName));
229 QString baseFileName = info.baseName();
230
231 // replace any spaces in filename with underscore to pevent
232 // errors when opening file in Linux through the command line.
233 baseFileName.replace(" ", "_");
234
235 QString path = qApp->Settings()->getBackupFilePath();
236 if (!info.exists(path))
237 {
238 path = info.absoluteDir().absolutePath();
239 }
240 QString backupFileName;
241 backupFileName = QString("%1/%2%3.%4").arg(path, baseFileName, "_(backup)", info.completeSuffix());
242
243 if (!SafeCopy(m_convertedFileName, backupFileName, error))
244 {
245#ifdef Q_OS_WIN32
246 qt_ntfs_permission_lookup++; // turn checking on
247#endif /*Q_OS_WIN32*/
248 const bool isFileWritable = info.isWritable();
249#ifdef Q_OS_WIN32
250 qt_ntfs_permission_lookup--; // turn it off again
251#endif /*Q_OS_WIN32*/
252
253 if (!isReadOnly() && isFileWritable)
254 {
255 const QString errorMsg(tr("Error creating a reserv copy: %1.").arg(error));
256 throw VException(errorMsg);
257 }
258 }
259 }

282void VAbstractConverter::BiasTokens(int position, int bias, QMap<int, QString> &tokens)
283{
284 QMap<int, QString> newTokens;
285 QMap<int, QString>::const_iterator i = tokens.constBegin();
286 while (i != tokens.constEnd())
287 {
288 if (i.key()<= position)
289 { // Tokens before position "position" did not change his positions.
290 newTokens.insert(i.key(), i.value());
291 }
292 else
293 {
294 newTokens.insert(i.key()-bias, i.value());
295 }
296 ++i;
297 }
298 tokens = newTokens;
299}

320//---------------------------------------------------------------------------------------------------------------------
321void VAbstractConverter::ValidateInputFile(const QString &currentSchema) const
322{
323 QString schema;
324 try
325 {
326 schema = getSchema(m_ver);
327 }
328 catch(const VException &error)
329 {
330 if (m_ver < minVer())
331 { // Version less than minimally supported version. Can't do anything.
332 throw;
333 }
334 else if (m_ver > maxVer())
335 { // Version bigger than maximum supported version. We still have a chance to open the file.
336 try
337 { // Try to open like the current version.
338 ValidateXML(currentSchema, m_convertedFileName);
339 }
340 catch(const VException &exp)
341 { // Nope, we can't.
342 Q_UNUSED(exp)
343 throw error;
344 }
345 }
346 else
347 { // Unexpected version. Most time mean that we do not catch all versions between min and max.
348 throw;
349 }
350
351 return; // All is fine and we can try to convert to current max version.
352 }
353
354 ValidateXML(schema, m_convertedFileName);
355}
