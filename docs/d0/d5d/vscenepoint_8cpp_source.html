85void VScenePoint::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
86{
87 const qreal scale = sceneScale(scene());
88
89 setPointPen(scale);
90 scaleCircleSize(this, scale * .75);
91
92 if (qApp->Settings()->getPointNameSize()*scale < 6 || !qApp->Settings()->getHidePointNames())
93 {
94 m_pointName->setVisible(false);
95 m_pointLeader->setVisible(false);
96 }
97 else
98 {
99 if (!m_onlyPoint)
100 {
101 m_pointName->setVisible(m_showPointName);
102
103 QPen leaderPen = m_pointLeader->pen();
104 QColor leaderColor = correctColor(m_pointLeader, QColor(qApp->Settings()->getPointNameColor()));
105 leaderColor.setAlpha(128);
106
107 if (qApp->Settings()->getUseToolColor())
108 {
109 leaderColor = correctColor(m_pointLeader, m_pointColor);
110 }
111
112 leaderPen.setColor(leaderColor);
113 m_pointLeader->setPen(leaderPen);
114
115 refreshLeader();
116 }
117 }
118
119 QGraphicsEllipseItem::paint(painter, option, widget);
120}

123void VScenePoint::refreshPointGeometry(const VPointF &point)
124{
125 setFlag(QGraphicsItem::ItemSendsGeometryChanges, false);
126 setPos(static_cast<QPointF>(point));
127 setFlag(QGraphicsItem::ItemSendsGeometryChanges, true);
128
129 m_showPointName = point.isShowPointName();
130
131 m_pointName->blockSignals(true);
132 m_pointName->setFlag(QGraphicsItem::ItemSendsGeometryChanges, false);
133 m_pointName->setPosition(QPointF(point.mx(), point.my()));
134 m_pointName->setFlag(QGraphicsItem::ItemSendsGeometryChanges, true);
135 m_pointName->blockSignals(false);
136 m_pointName->setText(point.name());
137 m_pointName->setTextColor(m_pointColor);
138 m_pointName->setVisible(m_showPointName);
139
140 refreshLeader();
141}

177void VScenePoint::refreshLeader()
178{
179 QRectF nRect = m_pointName->sceneBoundingRect();
180 nRect.translate(- scenePos());
181 if (!rect().intersects(nRect))
182 {
183 const QRectF nameRect = m_pointName->sceneBoundingRect();
184 QPointF p1;
185 QPointF p2;
186 VGObject::LineIntersectCircle(QPointF(), scaledRadius(sceneScale(scene())),
187 QLineF(QPointF(), nameRect.center() - scenePos()), p1, p2);
188 const QPointF pointRect = VGObject::LineIntersectRect(nameRect, QLineF(scenePos(), nameRect.center()));
189
190 if (QLineF(p1, pointRect - scenePos()).length() <= ToPixel(4/qMax(1.0, sceneScale(scene())), Unit::Mm))
191 {
192 m_pointLeader->setVisible(false);
193 }
194 else
195 {
196 m_pointLeader->setLine(QLineF(p1, pointRect - scenePos()));
197 m_pointLeader->setVisible(m_showPointName);
198 }
199 }
200 else
201 {
202 m_pointLeader->setVisible(false);
203 }
204}

207void VScenePoint::setPointPen(qreal scale)
208{
209 const qreal width = scaleWidth(m_isHovered ? widthMainLine : widthHairLine, scale);
210
211 if (qApp->Settings()->getUseToolColor() || isOnlyPoint())
212 {
213 setPen(QPen(correctColor(this, m_pointColor), width));
214 if (!m_onlyPoint)
215 {
216 if (!qApp->Settings()->isWireframe())
217 {
218 setBrush(QBrush(correctColor(this, m_pointColor), Qt::SolidPattern));
219 }
220 else
221 {
222 setBrush(QBrush(correctColor(this, m_pointColor), Qt::NoBrush));
223 }
224 }
225 else
226 {
227 setBrush(QBrush(correctColor(this, m_pointColor),Qt::NoBrush));
228 }
229 }
230 else
231 {
232 setPen(QPen(correctColor(this, QColor(qApp->Settings()->getPointNameColor())), width));
233
234 if (!qApp->Settings()->isWireframe())
235 {
236 setBrush(QBrush(correctColor(this, QColor(qApp->Settings()->getPointNameColor())),Qt::SolidPattern));
237 }
238 else
239 {
240 setBrush(QBrush(correctColor(this, QColor(qApp->Settings()->getPointNameColor())),Qt::NoBrush));
241 }
