80void TST_MeasurementRegExp::initTestCase()
81{
82 if (m_systemCode > systemCounts)
83 {
84 QFAIL("Unexpected system code.");
85 }
86
87 if (m_locale.isEmpty())
88 {
89 QFAIL("Empty locale code.");
90 }
91
92 const QStringList locales = SupportedLocales();
93
94 if (not locales.contains(m_locale))
95 {
96 QFAIL("Unsupported locale code.");
97 }
98
99 if (loadMeasurements(m_locale) != NoError)
100 {
101 const QString message = QString("Couldn't load measurements. Locale = %1")
102 .arg(m_locale);
103 QSKIP(qUtf8Printable(message));
104 }
105
106 if (LoadVariables(m_locale) != NoError)
107 {
108 const QString message = QString("Couldn't load variables. Locale = %1")
109 .arg(m_locale);
110 QSKIP(qUtf8Printable(message));
111 }
112
113 InitTrMs();//Very important do this after loading QM files.
114}

192int TST_MeasurementRegExp::loadMeasurements(const QString &checkedLocale)
193{
194 const QString path = TranslationsPath();
195 const QString file = QString("measurements_%1.qm").arg(checkedLocale);
196
197 if (QFileInfo(path+QLatin1String("/")+file).size() <= 34)
198 {
199 const QString message = QString("Translation for locale = %1 is empty. \nFull path: %2/%3")
200 .arg(checkedLocale)
201 .arg(path)
202 .arg(file);
203 QWARN(qUtf8Printable(message));
204
205 return ErrorSize;
206 }
207
208 delete m_pmsTranslator;
209 m_pmsTranslator = new QTranslator(this);
210
211 if (not m_pmsTranslator->load(file, path))
212 {
213 const QString message = QString("Can't load translation for locale = %1. \nFull path: %2/%3")
214 .arg(checkedLocale)
215 .arg(path)
216 .arg(file);
217 QWARN(qUtf8Printable(message));
218
219 delete m_pmsTranslator;
220
221 return ErrorLoad;
222 }
223
224 if (not QCoreApplication::installTranslator(m_pmsTranslator))
225 {
226 const QString message = QString("Can't install translation for locale = %1. \nFull path: %2/%3")
227 .arg(checkedLocale)
228 .arg(path)
229 .arg(file);
230 QWARN(qUtf8Printable(message));
231
232 delete m_pmsTranslator;
233
234 return ErrorInstall;
235 }
236
237 return NoError;
238}
