65/**
66 * @brief QmuTokenParser class constructor. Make easy initialization math parser.

71 *
72 * QmuTokenParser *cal = new QmuTokenParser(formula, osSeparator, false);
73 * tokens = cal->GetTokens();
74 * numbers = cal->GetNumbers();
75 * delete cal;
76 *
77 * @param formula string with formula.
78 * @param fromUser true if we parse formula from user

100/**
101 * @brief IsSingle test formula and return true if it contain only one number.
102 *
103 * Work only with expressions in internal (C) locale.
104 * @param formula expression for test
105 * @return true if fomula has single number
106 */
107bool QmuTokenParser::IsSingle(const QString &formula)
108{
109 if (formula.isEmpty())
110 {
111 return false;// if don't know say no
112 }
113
114 QScopedPointer<QmuTokenParser> cal(new QmuTokenParser());
115
116 // Parser doesn't know any variable on this stage. So, we just use variable factory that for each unknown
117 // variable set value to 0.
118 cal->SetVarFactory(AddVariable, cal.data());
119 cal->SetSepForEval();//Reset separators options
120
121 try
122 {
123 cal->SetExpr(formula);
124 cal->Eval();// We don't need save result, only parse formula
125 }
126 catch (const qmu::QmuParserError &error)
127 {
128 Q_UNUSED(error)
129 return false;// something wrong with formula, say no
130 }
131
132 QMap<int, QString> tokens = cal->GetTokens();// Tokens (variables, measurements)
133 const QMap<int, QString> numbers = cal->GetNumbers();// All numbers in expression
134 delete cal.take();
135
136 // Remove "-" from tokens list if exist. If don't do that unary minus operation will broken.
137 RemoveAll(tokens, QStringLiteral("-"));
138
139 if (tokens.isEmpty() && numbers.size() == 1)
140 {
141 return true;
142 }
143 else
144 {
145 return false;
146 }
147}
