112QSizeF AbstractLayoutDialog::RoundTemplateSize(qreal width, qreal height, Unit unit) const
113{
114 qreal w = 0;
115 qreal h = 0;
116
117 switch (unit)
118 {
119 case Unit::Cm:
120 case Unit::Mm:
121 case Unit::Px:
122 w = qRound(width * 100.0) / 100.0;
123 h = qRound(height * 100.0) / 100.0;
124 return QSizeF(w, h);
125 case Unit::Inch:
126 w = qRound(width * 100000.0) / 100000.0;
127 h = qRound(height * 100000.0) / 100000.0;
128 return QSizeF(w, h);
129 default:
130 break;
131 }
132
133 return QSizeF(width, height);
134}
