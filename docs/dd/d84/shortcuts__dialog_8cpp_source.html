399void ShortcutsDialog::exportPdf()
400{
401 QString filename = QFileDialog::getSaveFileName(nullptr, tr("Export PDF"), QString(),
402 "*.pdf", nullptr, QFileDialog::DontUseNativeDialog);
403
404 if (QFileInfo(filename).suffix().isEmpty())
405 {
406 filename.append(".pdf");
407 }
408 QPrinter printer(QPrinter::PrinterResolution);
409 printer.setOutputFormat(QPrinter::PdfFormat);
410 printer.setPageSize(printer.pageLayout().pageSize());
411 printer.setOutputFileName(filename);
412
413 QTextDocument textDocument;
414 textDocument.setHtml(ui->shortcuts_TextBrowser->toHtml());
415 textDocument.setPageSize(printer.pageLayout().paintRectPixels(printer.resolution()).size());
416 textDocument.print(&printer);
417}
