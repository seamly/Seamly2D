80void SavePiecePathOptions::undo()
81{
82 qCDebug(vUndo, "Undo.");
83
84 QDomElement domElement = doc->elementById(nodeId, VAbstractPattern::TagPath);
85 if (domElement.isElement())
86 {
87 VToolInternalPath::AddAttributes(doc, domElement, nodeId, m_oldPath);
88 doc->RemoveAllChildren(domElement);//Very important to clear before rewrite
89 VToolInternalPath::AddNodes(doc, domElement, m_oldPath);
90
91 IncrementReferences(m_oldPath.MissingNodes(m_newPath));
92
93 SCASSERT(m_data);
94 m_data->UpdatePiecePath(nodeId, m_oldPath);
95 }
96 else
97 {
98 qCWarning(vUndo, "Can't find path with id = %u.", nodeId);
99 return;
100 }
101}

128bool SavePiecePathOptions::mergeWith(const QUndoCommand *command)
129{
130 const SavePiecePathOptions *saveCommand = static_cast<const SavePiecePathOptions *>(command);
131 SCASSERT(saveCommand != nullptr);
132 const quint32 id = saveCommand->PathId();
133
134 if (id != nodeId)
135 {
136 return false;
137 }
138
139 m_newPath = saveCommand->NewPath();
140 return true;
141}
