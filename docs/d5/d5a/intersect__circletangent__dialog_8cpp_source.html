219//---------------------------------------------------------------------------------------------------------------------
220void IntersectCircleTangentDialog::ChosenObject(quint32 id, const SceneObject &type)
221{
222 if (prepare == false)// After first choose we ignore all objects
223 {
224 if (type == SceneObject::Point)
225 {
226 IntersectCircleTangentVisual *point = qobject_cast<IntersectCircleTangentVisual *>(vis);
227 SCASSERT(point != nullptr)
228
229 switch (number)
230 {
231 case 0:
232 if (SetObject(id, ui->comboBoxTangentPoint, tr("Select a circle center")))
233 {
234 number++;
235 point->VisualMode(id);
236 }
237 break;
238 case 1:
239 if (getCurrentObjectId(ui->comboBoxTangentPoint) != id)
240 {
241 if (SetObject(id, ui->comboBoxCircleCenter, ""))
242 {
243 number = 0;
244 point->setObject2Id(id);
245 point->RefreshGeometry();
246 prepare = true;
247 this->setModal(true);
248 this->show();
249 }
250 }
251 break;
252 default:
253 break;
254 }
255 }
256 }
257}

307//---------------------------------------------------------------------------------------------------------------------
308void IntersectCircleTangentDialog::EvalCircleRadius()
309{
310 labelEditFormula = ui->labelEditRadius;
311 const QString postfix = UnitsToStr(qApp->patternUnit(), true);
312 const qreal radius = Eval(ui->plainTextEditRadius->toPlainText(), flagCircleRadius,
313 ui->labelResultCircleRadius, postfix);
314
315 if (radius < 0)
316 {
317 flagCircleRadius = false;
318 ChangeColor(labelEditFormula, Qt::red);
319 ui->labelResultCircleRadius->setText(tr("Error"));
320 ui->labelResultCircleRadius->setToolTip(tr("Radius can't be negative"));
321
322 IntersectCircleTangentDialog::CheckState();
323 }
324}
