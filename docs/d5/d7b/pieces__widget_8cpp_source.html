185void PiecesWidget::togglePiece(quint32 id)
186{
187 ui->tableWidget->setSortingEnabled(false);
188 ui->tableWidget->clearSelection();
189
190 int selectedRow = 0;
191 const QHash<quint32, VPiece> *pieces = m_data->DataPieces();
192
193 if (pieces->contains(id))
194 {
195 for (int row = 0; row < ui->tableWidget->rowCount(); ++row)
196 {
197 QTableWidgetItem *item = ui->tableWidget->item(row, 0);
198 SCASSERT(item != nullptr)
199
200 if (item && item->data(Qt::UserRole).toUInt() == id)
201 {
202 selectedRow = row;
203 VPiece piece = pieces->value(id);
204 const bool inLayout = piece.isInLayout();
205 inLayout ? item->setIcon(QIcon("://icon/32x32/visible_on.png"))
206 : item->setIcon(QIcon("://icon/32x32/visible_off.png"));
207 {
208 QTableWidgetItem *item = ui->tableWidget->item(row, 1);
209 const bool locked = piece.isLocked();
210 locked ? item->setIcon(QIcon("://icon/32x32/lock_on.png"))
211 : item->setIcon(QIcon("://icon/32x32/lock_off.png"));
212 }
213
214 {
215 QTableWidgetItem *item = ui->tableWidget->item(row, 2);
216 QPixmap pixmap(20, 20);
217 pixmap.fill(QColor(piece.getColor()));
218 item->setIcon(QIcon(pixmap));
219 item->setData(Qt::UserRole, piece.getColor());
220 }
221
222 {
223 QTableWidgetItem *item = ui->tableWidget->item(row, 3);
224 item->setText(formatLetterString(piece));
225 }
226
227 {
228 QTableWidgetItem *item = ui->tableWidget->item(row, 4);
229 item->setText(piece.GetName());
230 }
231 }
232 }
233 }
234
235 ui->tableWidget->selectRow(selectedRow);
236 ui->tableWidget->setSortingEnabled(true);
237}

263void PiecesWidget::cellClicked(int row, int column)
264{
265 QTableWidgetItem *item = ui->tableWidget->item(row, 0);
266 if (!item) return;
267
268 const quint32 id = item->data(Qt::UserRole).toUInt();
269 const QHash<quint32, VPiece> *allPieces = m_data->DataPieces();
270 const bool locked = allPieces->value(id).isLocked();
271
272 if (column == 0)
273 {
274 if (locked == false)
275 {
276 const bool inLayout = !allPieces->value(id).isInLayout();
277
278 TogglePieceInLayout *command = new TogglePieceInLayout(id, inLayout, m_data, m_doc);
279 connect(command, &TogglePieceInLayout::updateList, this, &PiecesWidget::togglePiece);
280 qApp->getUndoStack()->push(command);
281 }
282 else
283 {
284 QApplication::beep();
285 }
286 }
287 else if (column == 1)
288 {
289 TogglePieceLock *command = new TogglePieceLock(id, !locked, m_data, m_doc);
290 connect(command, &TogglePieceLock::updateList, this, &PiecesWidget::togglePiece);
291 qApp->getUndoStack()->push(command);
292
293 VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
294 SCASSERT(scene != nullptr)
295 emit scene->pieceLockedChanged(id, locked);
296 }
297 if (column == 2)
298 {
299 if (locked == true)
300 {
301 QApplication::beep();
302 }
303 }
304 else if (column == 3 || column == 4)
305 {
306 if (locked == true)
307 {
308 QApplication::beep();
309 }
310
311 }
312 emit Highlight(id);
313}

346void PiecesWidget::fillTable(const QHash<quint32, VPiece> *pieces)
347{
348 ui->tableWidget->blockSignals(true);
349
350 const int selectedRow = ui->tableWidget->currentRow();
351 ui->tableWidget->clearContents();
352 ui->tableWidget->setColumnCount(5);
353 ui->tableWidget->setRowCount(pieces->size());
354 ui->tableWidget->setSortingEnabled(false);
355
356 qint32 currentRow = -1;
357 auto i = pieces->constBegin();
358 while (i != pieces->constEnd())
359 {
360 ++currentRow;
361 const VPiece piece = i.value();
362
363 // Add in layout item
364 PieceTableWidgetItem *item = new PieceTableWidgetItem(m_data);
365 item->setTextAlignment(Qt::AlignHCenter);
366 item->setSizeHint(QSize(20, 20));
367 item->setIcon(piece.isInLayout() ? QIcon("://icon/32x32/visible_on.png") : QIcon("://icon/32x32/visible_off.png"));
368 item->setData(Qt::UserRole, i.key());
369 item->setFlags(item->flags() &= ~(Qt::ItemIsEditable)); // set the item non-editable (view only), and non-selectable
370 item->setToolTip(tr("Toggle inclusion of pattern piece in layout"));
371 ui->tableWidget->setItem(currentRow, 0, item);
372
373 // Add locked item
374 item = new PieceTableWidgetItem(m_data);
375 item->setTextAlignment(Qt::AlignHCenter);
376 item->setSizeHint(QSize(20, 20));
377 item->setIcon(piece.isLocked() ? QIcon("://icon/32x32/lock_on.png") : QIcon("://icon/32x32/lock_off.png"));
378 item->setData(Qt::UserRole, i.key());
379 item->setFlags(item->flags() &= ~(Qt::ItemIsEditable)); // set the item non-editable (view only), and non-selectable
380 item->setToolTip(tr("Toggle lock on pattern piece"));
381 ui->tableWidget->setItem(currentRow, 1, item);
382
383 // Add color item
384 item = new PieceTableWidgetItem(m_data);
385 item->setTextAlignment(Qt::AlignLeft | Qt::AlignVCenter);
386 item->setSizeHint(QSize(20, 20));
387 QPixmap pixmap(20, 20);
388 pixmap.fill(QColor(piece.getColor()));
389 item->setIcon(QIcon(pixmap));
390 item->setData(Qt::UserRole, piece.getColor());
391 item->setFlags(item->flags() &= ~(Qt::ItemIsEditable)); // set the item non-editable (view only), and non-selectable
392 item->setToolTip(tr("Double click opens color selector"));
393 ui->tableWidget->setItem(currentRow, 2, item);
394
395 { // Add letter item
396 QTableWidgetItem *item = new QTableWidgetItem(formatLetterString(piece));
397 item->setTextAlignment(Qt::AlignHCenter | Qt::AlignVCenter);
398 item->setFlags(item->flags() &= ~(Qt::ItemIsEditable)); // set the item non-editable (view only), and non-selectable
399 item->setToolTip(tr("Double click opens pattern piece properties dialog"));
400 ui->tableWidget->setItem(currentRow, 3, item);
401 }
402
403 { // Add name item
404 QTableWidgetItem *item = new QTableWidgetItem(piece.GetName());
405 item->setTextAlignment(Qt::AlignLeft | Qt::AlignVCenter);
406 item->setFlags(item->flags() &= ~(Qt::ItemIsEditable)); // set the item non-editable (view only), and non-selectable
407 item->setToolTip(tr("Double click opens pattern piece properties dialog"));
408 ui->tableWidget->setItem(currentRow, 4, item);
409 }
410
411 ++i;
412 }
413
414 ui->tableWidget->setHorizontalHeaderItem(0, new QTableWidgetItem(makeHeaderName(tr("Included"))));
415 ui->tableWidget->horizontalHeaderItem(0)->setToolTip(tr("Pattern piece is included in layout"));
416 ui->tableWidget->setHorizontalHeaderItem(1, new QTableWidgetItem(makeHeaderName(tr("Locked"))));
417 ui->tableWidget->horizontalHeaderItem(1)->setToolTip(tr("Pattern piece is locked"));
418 ui->tableWidget->setHorizontalHeaderItem(2, new QTableWidgetItem(makeHeaderName(tr("Color"))));
419 ui->tableWidget->horizontalHeaderItem(2)->setToolTip(tr("Pattern piece color"));
420 ui->tableWidget->setHorizontalHeaderItem(3, new QTableWidgetItem(makeHeaderName(tr("Piece"))));
421 ui->tableWidget->horizontalHeaderItem(3)->setToolTip(tr("Pattern piece letter"));
422 ui->tableWidget->setHorizontalHeaderItem(4, new QTableWidgetItem(tr("Name")));
423 ui->tableWidget->horizontalHeaderItem(4)->setToolTip(tr("Pattern piece name"));
424 ui->tableWidget->horizontalHeaderItem(4)->setTextAlignment(Qt::AlignLeft | Qt::AlignVCenter);
425 ui->tableWidget->horizontalHeader()->setSectionResizeMode(QHeaderView::Fixed);
426 ui->tableWidget->resizeColumnsToContents();
427 ui->tableWidget->resizeRowsToContents();
428 //ui->tableWidget->setCurrentCell(selectedRow, 0);
429 ui->tableWidget->setSortingEnabled(true);
430 //ui->tableWidget->clearSelection();
431 ui->tableWidget->blockSignals(false);
432}

500/**
501 * @brief formatLetterString makes name to include piece letter and piece name
502 * @param piece pattern piece
503 */
504QString PiecesWidget::formatLetterString(const VPiece piece)
505{
506 QRegularExpression regExp("^\\d+$");
507 const VPieceLabelData& pieceData = piece.GetPatternPieceData();
508 QString letter = pieceData.GetLetter();
509
510 if (!letter.isEmpty())
511 {
512 if (regExp.match(letter).hasMatch())
513 {
514 int number = letter.toInt();
515 letter = QStringLiteral("%1").arg(number, MAX_LENGTH, BASE_10, QLatin1Char('0'));
516 }
517 }
518 return letter;
519}

521void PiecesWidget::showContextMenu(const QPoint &pos)
522{
523 ui->tableWidget->horizontalHeader()->setSortIndicator(-1, Qt::AscendingOrder);
524 ui->tableWidget->setSortingEnabled(false);
525 ui->tableWidget->blockSignals(true);
526
527 // workaround for https://bugreports.qt.io/browse/QTBUG-97559: assign parent to QMenu
528 QScopedPointer<QMenu> menu(new QMenu(ui->tableWidget));
529 QAction *selectAll = menu->addAction(tr("Include all pieces"));
530 QAction *selectNone = menu->addAction(tr("Exclude all pieces"));
531 QAction *invertSelection = menu->addAction(tr("Invert included pieces"));
532
533 QAction *separator = new QAction(this);
534 separator->setSeparator(true);
535 menu->addAction(separator);
536
537 QAction *lockAll = menu->addAction(tr("Lock all pieces"));
538 QAction *unlockAll = menu->addAction(tr("Unlock all pieces"));
539 QAction *invertLocked = menu->addAction(tr("Invert locked pieces"));
540
541 m_allPieces = m_data->DataPieces();
542 if (m_allPieces->isEmpty())
543 {
544 return;
545 }
546
547 int selectedPieces = 0;
548 int lockedPieces = 0;
549
550 auto piece = m_allPieces->constBegin();
551 while (piece != m_allPieces->constEnd())
552 {
553 if(piece.value().isInLayout())
554 {
555 selectedPieces++;
556 }
557 if(piece.value().isLocked())
558 {
559 lockedPieces++;
560 }
561 ++piece;
562 }
563
564 if (selectedPieces == 0)
565 {
566 selectNone->setDisabled(true);
567 }
568 else if (selectedPieces == m_allPieces->size())
569 {
570 selectAll->setDisabled(true);
571 }
572
573 if (lockedPieces == 0)
574 {
575 unlockAll->setDisabled(true);
576 }
577 else if (lockedPieces == m_allPieces->size())
578 {
579 lockAll->setDisabled(true);
580 }
581
582 QAction *selectedAction = menu->exec(ui->tableWidget->viewport()->mapToGlobal(pos));
583
584 if (selectedAction == selectAll)
585 {
586 includeAllPieces();
587 }
588 else if (selectedAction == selectNone)
589 {
590 excludeAllPieces();
591 }
592 else if (selectedAction == invertSelection)
593 {
594 invertIncludedPieces();
595 }
596
597 else if (selectedAction == lockAll)
598 {
599 lockAllPieces();
600 }
601 else if (selectedAction == unlockAll)
602 {
603 unlockAllPieces();
604 }
605 else if (selectedAction == invertLocked)
606 {
607 invertLockedPieces();
608 }
609 ui->tableWidget->setSortingEnabled(true);
610 ui->tableWidget->blockSignals(false);
611}
