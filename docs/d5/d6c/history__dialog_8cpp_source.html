95 //Limit dialog height to 80% of screen size
96 setMaximumHeight(qRound(QGuiApplication::primaryScreen()->availableGeometry().height() * .8));
97
98 ui->find_LineEdit->installEventFilter(this);
99
100 qApp->Settings()->getOsSeparator() ? setLocale(QLocale()) : setLocale(QLocale::c());
101
102 fillTable();
103 initializeTable();
104
105 ok_Button = ui->buttonBox->button(QDialogButtonBox::Ok);
106 connect(ok_Button, &QPushButton::clicked, this, &HistoryDialog::DialogAccepted);
107 connect(ui->clipboard_ToolButton, &QToolButton::clicked, this, &HistoryDialog::copyToClipboard);
108 connect(ui->tableWidget, &QTableWidget::cellClicked, this, &HistoryDialog::cellClicked);
109 connect(m_doc, &VPattern::ChangedCursor, this, &HistoryDialog::changedCursor);

139/**
140 * @brief cellClicked changed history record
141 * @param row number row in table
142 * @param column number column in table
143 */
144void HistoryDialog::cellClicked(int row, int column)
145{
146 if (column == 0)
147 {
148 QTableWidgetItem *item = ui->tableWidget->item(m_cursorRow, 0);
149 item->setIcon(QIcon());
150
151 item = ui->tableWidget->item(row, 0);
152 m_cursorRow = row;
153 item->setIcon(QIcon("://icon/24x24/left_to_right_arrow.png"));
154 const quint32 id = qvariant_cast<quint32>(item->data(Qt::UserRole));
155 m_doc->blockSignals(true);
156 row == ui->tableWidget->rowCount()-1 ? m_doc->setCursor(0) : m_doc->setCursor(id);
157 m_doc->blockSignals(false);
158 }
159 else
160 {
161 QTableWidgetItem *item = ui->tableWidget->item(m_cursorToolRecordRow, 0);
162 quint32 id = qvariant_cast<quint32>(item->data(Qt::UserRole));
163 emit showHistoryTool(id, false);
164
165 m_cursorToolRecordRow = row;
166 item = ui->tableWidget->item(m_cursorToolRecordRow, 0);
167 id = qvariant_cast<quint32>(item->data(Qt::UserRole));
168 emit showHistoryTool(id, true);
169 }
170}

204/**
205 * @brief fillTable fill table
206 */
207void HistoryDialog::fillTable()
208{
209 ui->tableWidget->clear();
210 QVector<VToolRecord> history = m_doc->getBlockHistory();
211 qint32 currentRow = -1;
212 qint32 count = 0;
213 ui->tableWidget->setRowCount(history.size());//Set Row count to number of Tool history records
214
215 for (qint32 i = 0; i< history.size(); ++i)
216 {
217 const VToolRecord tool = history.at(i);
218 const RowData rowData = record(tool);
219 if (!rowData.tool.isEmpty())
220 {
221 currentRow++;
222
223 {
224 QTableWidgetItem *item = new QTableWidgetItem(QString().setNum(rowData.id));
225 item->setTextAlignment(Qt::AlignHCenter);
226 item->setSizeHint(QSize(80, 24));
227 item->setData(Qt::UserRole, rowData.id);
228 item->setFlags(item->flags() ^ Qt::ItemIsEditable);
229 ui->tableWidget->setItem(currentRow, 0, item);//2nd column is Tool id
230 }
231
232 {
233 QTableWidgetItem *item = new QTableWidgetItem(QString(rowData.name));
234 item->setTextAlignment(Qt::AlignHCenter);
235 item->setSizeHint(QSize(100, 24));
236 item->setFlags(item->flags() ^ Qt::ItemIsEditable);
237 ui->tableWidget->setItem(currentRow, 1, item);//3rd column is Tool name
238 }
239
240 QTableWidgetItem *item = new QTableWidgetItem(rowData.tool);
241 item->setTextAlignment(Qt::AlignLeft);
242 item->setFlags(item->flags() ^ Qt::ItemIsEditable);
243 item->setIcon(QIcon(rowData.icon));
244 ui->tableWidget->setItem(currentRow, 2, item);//4th column is Tool description
245 ++count;
246 }
247 }
248 ui->tableWidget->setRowCount(count);//Current history records count
249 if (count>0)
250 {
251 m_cursorRow = cursorRow();
252 QTableWidgetItem *item = ui->tableWidget->item(m_cursorRow, 0);
253 SCASSERT(item != nullptr)
254 item->setIcon(QIcon("://icon/24x24/left_to_right_arrow.png"));//place arrow in 1st column of most recent history record
255 }
256
257 ui->tableWidget->resizeColumnsToContents();
258 ui->tableWidget->resizeRowsToContents();
259 ui->tableWidget->verticalHeader()->setDefaultSectionSize(24);//Set all row heights to 20px
260}

783void HistoryDialog::copyToClipboard()
784{
785 QItemSelectionModel *model = ui->tableWidget->selectionModel();
786 QModelIndexList selectedIndexes = model->selectedIndexes();
787
788 QString clipboardString;
789
790 for (int i = 0; i < selectedIndexes.count(); ++i)
791 {
792 QModelIndex current = selectedIndexes[i];
793 QString displayText = current.data(Qt::DisplayRole).toString();
794
795 // Check if another column exists beyond this one.
796 if (i + 1 < selectedIndexes.count())
797 {
798 QModelIndex next = selectedIndexes[i+1];
799
800 // If the column is on different row, the clipboard should take note.
801 if (next.row() != current.row())
802 {
803 displayText.append("\n");
804 }
805 else
806 {
807 // Otherwise append a comma separator.
808 displayText.append(" , ");
809 }
810 }
811 clipboardString.append(displayText);
812 }
813
814 QClipboard *clipboard = QApplication::clipboard();
815 clipboard->setText(clipboardString);
816}
