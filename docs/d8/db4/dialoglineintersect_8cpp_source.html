120/**
121 * @brief ChosenObject gets id and type of selected object. Save right data and ignore wrong.
122 * @param id id of point or detail
123 * @param type type of object
124 */
125void DialogLineIntersect::ChosenObject(quint32 id, const SceneObject &type)
126{
127 if (prepare == false)// After first choose we ignore all objects
128 {
129 if (type == SceneObject::Point)
130 {
131 VisToolLineIntersect *line = qobject_cast<VisToolLineIntersect *>(vis);
132 SCASSERT(line != nullptr)
133
134 switch (number)
135 {
136 case 0:
137 if (SetObject(id, ui->comboBoxP1Line1, tr("Select second point of first line")))
138 {
139 number++;
140 line->VisualMode(id);
141 }
142 break;
143 case 1:
144 if (getCurrentObjectId(ui->comboBoxP1Line1) != id)
145 {
146 if (SetObject(id, ui->comboBoxP2Line1, tr("Select first point of second line")))
147 {
148 number++;
149 line->setLine1P2Id(id);
150 line->RefreshGeometry();
151 }
152 }
153 break;
154 case 2:
155 if (SetObject(id, ui->comboBoxP1Line2, tr("Select second point of second line")))
156 {
157 number++;
158 line->setLine2P1Id(id);
159 line->RefreshGeometry();
160 }
161 break;
162 case 3:
163 {
164 QSet<quint32> set;
165 set.insert(getCurrentObjectId(ui->comboBoxP1Line1));
166 set.insert(getCurrentObjectId(ui->comboBoxP2Line1));
167 set.insert(getCurrentObjectId(ui->comboBoxP1Line2));
168 set.insert(id);
169
170 if (set.size() >= 3)
171 {
172 if (SetObject(id, ui->comboBoxP2Line2, ""))
173 {
174 line->setLine2P2Id(id);
175 line->RefreshGeometry();
176 prepare = true;
177 flagPoint = CheckIntersecion();
178 CheckState();
179 if (flagPoint)
180 {
181 DialogAccepted();
182 }
183 else
184 {
185 this->setModal(true);
186 this->show();
187
188 connect(ui->comboBoxP1Line1, &QComboBox::currentTextChanged, this,
189 &DialogLineIntersect::PointChanged);
190 connect(ui->comboBoxP2Line1, &QComboBox::currentTextChanged, this,
191 &DialogLineIntersect::PointChanged);
192 connect(ui->comboBoxP1Line2, &QComboBox::currentTextChanged, this,
193 &DialogLineIntersect::PointChanged);
194 connect(ui->comboBoxP2Line2, &QComboBox::currentTextChanged, this,
195 &DialogLineIntersect::PointChanged);
196 }
197 }
198 }
199 }
200 break;
201 default:
202 break;
203 }
204 }
205 }
206}

234void DialogLineIntersect::PointNameChanged()
235{
236 QSet<quint32> set;
237 const quint32 p1Line1Id = getCurrentObjectId(ui->comboBoxP1Line1);
238 const quint32 p2Line1Id = getCurrentObjectId(ui->comboBoxP2Line1);
239 const quint32 p1Line2Id = getCurrentObjectId(ui->comboBoxP1Line2);
240 const quint32 p2Line2Id = getCurrentObjectId(ui->comboBoxP2Line2);
241
242 set.insert(p1Line1Id);
243 set.insert(p2Line1Id);
244 set.insert(p1Line2Id);
245 set.insert(p2Line2Id);
246
247 const QSharedPointer<VPointF> p1Line1 = data->GeometricObject<VPointF>(p1Line1Id);
248 const QSharedPointer<VPointF> p2Line1 = data->GeometricObject<VPointF>(p2Line1Id);
249 const QSharedPointer<VPointF> p1Line2 = data->GeometricObject<VPointF>(p1Line2Id);
250 const QSharedPointer<VPointF> p2Line2 = data->GeometricObject<VPointF>(p2Line2Id);
251
252 QLineF line1(static_cast<QPointF>(*p1Line1), static_cast<QPointF>(*p2Line1));
253 QLineF line2(static_cast<QPointF>(*p1Line2), static_cast<QPointF>(*p2Line2));
254 QPointF fPoint;
255 QLineF::IntersectType intersect = line1.intersects(line2, &fPoint);
256
257 QColor color = okColor;
258 if (set.size() < 3 || intersect == QLineF::NoIntersection)
259 {
260 flagError = false;
261 color = errorColor;
262 }
263 else
264 {
265 flagError = true;
266 color = okColor;
267 }
268 ChangeColor(ui->labelP1Line1, color);
269 ChangeColor(ui->labelP2Line1, color);
270 ChangeColor(ui->labelP1Line2, color);
271 ChangeColor(ui->labelP2Line2, color);
272 CheckState();
273}

292/**
293 * @brief CheckIntersecion check intersection of points
294 * @return true - line have intersection, false = don't have
295 */
296bool DialogLineIntersect::CheckIntersecion()
297{
298 const QSharedPointer<VPointF> p1L1 = data->GeometricObject<VPointF>(GetP1Line1());
299 const QSharedPointer<VPointF> p2L1 = data->GeometricObject<VPointF>(GetP2Line1());
300 const QSharedPointer<VPointF> p1L2 = data->GeometricObject<VPointF>(GetP1Line2());
301 const QSharedPointer<VPointF> p2L2 = data->GeometricObject<VPointF>(GetP2Line2());
302
303 QLineF line1(static_cast<QPointF>(*p1L1), static_cast<QPointF>(*p2L1));
304 QLineF line2(static_cast<QPointF>(*p1L2), static_cast<QPointF>(*p2L2));
305 QPointF fPoint;
306 QLineF::IntersectType intersect = line1.intersects(line2, &fPoint);
307 if (intersect == QLineF::UnboundedIntersection || intersect == QLineF::BoundedIntersection)
308 {
309 return true;
310 }
311 else
312 {
313 return false;
314 }
315}
