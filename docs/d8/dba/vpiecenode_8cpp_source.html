64qreal EvalFormula(const VContainer *data, QString formula)
65{
66 if (formula.isEmpty())
67 {
68 return -1;
69 }
70 else
71 {
72 try
73 {
74 // Replace line return character with spaces for calc if exist
75 formula.replace("\n", " ");
76 QScopedPointer<Calculator> cal(new Calculator());
77 const qreal result = cal->EvalFormula(data->DataVariables(), formula);
78
79 if (qIsInf(result) || qIsNaN(result))
80 {
81 return -1;
82 }
83 return result;
84 }
85 catch (qmu::QmuParserError &error)
86 {
87 Q_UNUSED(error)
88 return -1;
89 }
90 }
91}
92}

200qreal VPieceNode::GetSABefore(const VContainer *data, Unit unit) const
201{
202 if (d->m_beforeWidthFormula == currentSeamAllowance)
203 {
204 return -1;
205 }
206
207 qreal value = EvalFormula(data, d->m_beforeWidthFormula);
208 if (value >= 0)
209 {
210 value = ToPixel(value, unit);
211 }
212 return value;
