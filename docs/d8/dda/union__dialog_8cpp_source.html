138//---------------------------------------------------------------------------------------------------------------------
139bool UnionDialog::checkPiece(const quint32 &pieceId) const
140{
141 if (pieceId == NULL_ID)
142 {
143 return false;
144 }
145 const VPiece piece = data->GetPiece(pieceId);
146 if (piece.isLocked())
147 {
148 ui->checkBox->setChecked(true);
149 ui->checkBox->setEnabled(false);
150 }
151 if (piece.GetPath().CountNodes() >= 3 && piece.GetPath().ListNodePoint().size() >= 2)
152 {
153 return true;
154 }
155 else
156 {
157 return false;
158 }
159}

162/**
163 * @brief chosenPiece help save information about piece and points on piece
164 * @param id id selected object
165 * @param type type selected object
166 * @param pieceId id of piece
167 * @param index index of edge
168 */
169void UnionDialog::chosenPiece(const quint32 &id, const SceneObject &type, quint32 &pieceId,
170 int &index)
171{
172 if (pieceId == NULL_ID)
173 {
174 if (type == SceneObject::Piece)
175 {
176 m_beep->play();
177 if (checkPiece(id))
178 {
179 pieceId = id;
180 emit ToolTip(tr("Select the first point"));
181 return;
182 }
183 else
184 {
185 emit ToolTip(tr("Pattern piece should have at least two points and three objects"));
186 return;
187 }
188 }
189 }
190 if (CheckObject(id, pieceId) == false)
191 {
192 return;
193 }
194 if (type == SceneObject::Point)
195 {
196 m_beep->play();
197 if (numberP == 0)
198 {
199 p1 = id;
200 ++numberP;
201 emit ToolTip(tr("Select a second point"));
202 return;
203 }
204 if (numberP == 1)
205 {
206 if (id == p1)
207 {
208 emit ToolTip(tr("Select a unique point"));
209 return;
210 }
211 VPiece piece = data->GetPiece(pieceId);
212 if (piece.GetPath().OnEdge(p1, id))
213 {
214 p2 = id;
215 index = piece.GetPath().Edge(p1, p2);
216 ++numberD;
217 if (numberD > 1)
218 {
219 ++numberP;
220 emit ToolTip("");
221 this->setModal(true);
222 this->show();
223 return;
224 }
225 else
226 {
227 numberP = 0;
228 p1 = 0;
229 p2 = 0;
230 emit ToolTip(tr("Select a piece"));
231 return;
232 }
233 }
234 else
235 {
236 emit ToolTip(tr("Select a point on edge"));
237 return;
238 }
239 }
240 }
