104//---------------------------------------------------------------------------------------------------------------------
105void ArrowedLineItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
106{
107 QPen linePen = pen();
108 linePen.setWidthF(scaleWidth(widthMainLine, sceneScale(scene())));
109 setPen(linePen);
110 m_arrows->setPen(linePen);
111
112 QPainterPath path;
113 path.moveTo(line().p1());
114 path.lineTo(line().p2());
115
116 qreal arrow_step = 60;
117 qreal arrow_size = 10;
118
119 if (line().length() < arrow_step)
120 {
121 drawArrow(line(), path, arrow_size);
122 }
123
124 QLineF axis;
125 axis.setP1(line().p1());
126 axis.setAngle(line().angle());
127 axis.setLength(arrow_step);
128
129 int steps = qFloor(line().length()/arrow_step);
130 for (int i=0; i<steps; ++i)
131 {
132 drawArrow(axis, path, arrow_size);
133 axis.setLength(axis.length()+arrow_step);
134 }
135 m_arrows->setPath(path);
136
137 QGraphicsLineItem::paint(painter, option, widget);
138}

140//---------------------------------------------------------------------------------------------------------------------
141void ArrowedLineItem::drawArrow(const QLineF &axis, QPainterPath &path, const qreal &arrow_size)
142{
143 QLineF arrowPart1;
144 arrowPart1.setP1(axis.p2());
145 arrowPart1.setLength(arrow_size);
146 arrowPart1.setAngle(axis.angle()+180+35);
147
148 path.moveTo(arrowPart1.p1());
149 path.lineTo(arrowPart1.p2());
150
151 QLineF arrowPart2;
152 arrowPart2.setP1(axis.p2());
153 arrowPart2.setLength(arrow_size);
154 arrowPart2.setAngle(axis.angle()+180-35);
155
156 path.moveTo(arrowPart2.p1());
157 path.lineTo(arrowPart2.p2());
158}
