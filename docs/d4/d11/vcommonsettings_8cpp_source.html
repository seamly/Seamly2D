261void SymlinkCopyDirRecursive(const QString &fromDir, const QString &toDir, bool replaceOnConflit)
262{
263 QDir dir;
264 dir.setPath(fromDir);
265
266 foreach (QString copyFile, dir.entryList(QDir::Files))
267 {
268 const QString from = fromDir + QDir::separator() + copyFile;
269 QString to = toDir + QDir::separator() + copyFile;
270
271#ifdef Q_OS_WIN
272 {
273 // To fix issue #702 check each not symlink if it is actually broken symlink.
274 // Also trying to mimic Unix symlink. If a file eaxists do not create a symlink and remove it if exists.
275 QFile fileTo(to);
276 if (fileTo.exists())
277 {
278 if (not fileTo.rename(to + QLatin1String(".lnk")))
279 {
280 QFile::remove(to + QLatin1String(".lnk"));
281 fileTo.rename(to + QLatin1String(".lnk"));
282 }
283
284 QFileInfo info(to + QLatin1String(".lnk"));
285 if (info.symLinkTarget().isEmpty())
286 {
287 fileTo.copy(to);
288 fileTo.remove();
289 continue; // The file already exists, skip creating shortcut
290 }
291 }
292 }
293
294 to = to + QLatin1String(".lnk");
295#endif
296
297 if (QFile::exists(to))
298 {
299 if (replaceOnConflit)
300 {
301 QFile::remove(to);
302 }
303 else
304 {
305 continue;
306 }
307 }
308
309 QFile::link(from, to);
310 }
311
312 foreach (QString copyDir, dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot))
313 {
314 const QString from = fromDir + QDir::separator() + copyDir;
315 const QString to = toDir + QDir::separator() + copyDir;
316
317 if (dir.mkpath(to) == false)
318 {
319 return;
320 }
321
322 SymlinkCopyDirRecursive(from, to, replaceOnConflit);
323 }
324}

327QString PrepareStandardFiles(const QString &currentPath, const QString &standardPath, const QString &defPath)
328{
329 QDir standardPathDir(standardPath);
330 QDir currentPathDir(currentPath);
331 if ((currentPath == defPath || not currentPathDir.exists()) && standardPathDir.exists())
332 {
333 const QDir localdata (defPath);
334 if (localdata.mkpath("."))
335 {
336 SymlinkCopyDirRecursive(standardPath, defPath, false);
337 }
338 return defPath;
339 }
340 return currentPath;
341}

352{

354 return QCoreApplication::applicationDirPath() + shareItem;
355#elif defined(Q_OS_MAC)
356 QDir dirBundle(QCoreApplication::applicationDirPath() + QStringLiteral("/../Resources") + shareItem);
357 if (dirBundle.exists())
358 {
359 return dirBundle.absolutePath();
360 }
361 else
362 {
363 QDir appDir = QDir(qApp->applicationDirPath());
364 appDir.cdUp();
365 appDir.cdUp();
366 appDir.cdUp();
367 QDir dir(appDir.absolutePath() + shareItem);
368 if (dir.exists())
369 {
370 return dir.absolutePath();
371 }
372 else
373 {
374 return VCommonSettings::unixStandardSharePath + shareItem;
375 }
376 }
377#else // Unix
378#ifdef QT_DEBUG
379 return QCoreApplication::applicationDirPath() + shareItem;
380#else
381 QDir dir(QCoreApplication::applicationDirPath() + shareItem);
382 if (dir.exists())
383 {
384 return dir.absolutePath();
385 }

388 return VCommonSettings::unixStandardSharePath + shareItem;
