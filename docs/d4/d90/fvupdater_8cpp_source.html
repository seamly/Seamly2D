139bool FvUpdater::CheckForUpdates(bool silentAsMuchAsItCouldGet) {
140 if (m_feedURL.isEmpty()) {
141 qCritical()
142 << "Please set feed URL via setFeedURL() before calling CheckForUpdates().";
143 return false;
144 }
145
146 m_silentAsMuchAsItCouldGet = silentAsMuchAsItCouldGet;
147
148 // Check if application's organization name and domain are set, fail
149 // otherwise (nowhere to store QSettings to)
150 if (QCoreApplication::organizationName().isEmpty()) {
151 qCritical()
152 << "QApplication::organizationName is not set. Please do that.";
153 return false;
154 }
155 if (QCoreApplication::organizationDomain().isEmpty()) {
156 qCritical()
157 << "QApplication::organizationDomain is not set. Please do that.";
158 return false;
159 }
160
161 // Set application name / version is not set yet
162 if (QCoreApplication::applicationName().isEmpty()) {
163 qCritical()
164 << "QApplication::applicationName is not set. Please do that.";
165 return false;
166 }
167
168 if (QCoreApplication::applicationVersion().isEmpty()) {
169 qCritical()
170 << "QApplication::applicationVersion is not set. Please do that.";
171 return false;
172 }
173
174 cancelDownloadFeed();
175 m_httpRequestAborted = false;
176 startDownloadFeed(m_feedURL);
177
178 return true;
179}

214void FvUpdater::startDownloadFile(QUrl url, QString name) {
215 QNetworkRequest request;
216 request.setHeader(QNetworkRequest::ContentTypeHeader,
217 QStringLiteral("application/text"));
218 request.setHeader(QNetworkRequest::UserAgentHeader,
219 QCoreApplication::applicationName());
220 request.setUrl(url);
221 request.setSslConfiguration(QSslConfiguration::defaultConfiguration());
222
223 m_reply = m_qnam.get(request);
224 connect(m_reply, SIGNAL(metaDataChanged()), this, SLOT(getFileSize()));
225 QDir downloadDir = QStandardPaths::writableLocation(QStandardPaths::DownloadLocation);
226 downloadDir.mkdir(m_releaseName);
227 downloadDir.cd(m_releaseName);
228 auto downloadedFile = new QFile(downloadDir.filePath(name), this);
229 if(downloadedFile->exists() && !downloadedFile->remove()){
230 showErrorDialog(tr("Unable to get exclusive access to file \n%1\nPossibly the file is already being downloaded.").arg(downloadDir.filePath(name)), false);
231 return;
232 }
233 bool isOpen = downloadedFile->open(QIODevice::WriteOnly | QIODevice::Truncate);
234 if (!isOpen) {
235 showErrorDialog(tr("Unable to open file\n%1\nfor writing").arg(downloadDir.filePath(name)), false);
236 return;
237 }
238 connect(m_reply.data(), &QNetworkReply::readyRead, this, [this, downloadedFile]() {
239 // this slot gets called every time the QNetworkReply has new data.
240 // We read all of its new data and write it into the file.
241 // That way we use less RAM than when reading it at the finished()
242 // signal of the QNetworkReply
243 downloadedFile->write(m_reply->readAll());
244 int progress = int(downloadedFile->size() * 100 / m_fileSize);
245 setProgress(progress);
246 });
247
248 connect(m_reply.data(), SIGNAL(error(QNetworkReply::NetworkError)), this, SLOT(networkError(QNetworkReply::NetworkError)));
249
250
251 connect(m_reply.data(), &QNetworkReply::downloadProgress, this,
252 [this](qint64 bytesRead, qint64 totalBytes) {
253 Q_UNUSED(bytesRead)
254 Q_UNUSED(totalBytes)
255
256 if (m_httpRequestAborted) {
257 return;
258 }
259 });
260 connect(m_reply.data(), &QNetworkReply::finished, this, [=]() {
261 fileDownloadFinished(downloadedFile, name);
262 });

264void FvUpdater::fileDownloadFinished(QFile *downloadedFile, QString name) {
265 if (m_httpRequestAborted) {
266 m_reply->deleteLater();
267 return;
268 }
269
270 const QVariant redirectionTarget =
271 m_reply->attribute(QNetworkRequest::RedirectionTargetAttribute);
272 if (m_reply->error() != QNetworkReply::NoError) {
273 // Error.
274 showErrorDialog(
275 tr("File download failed: %1.").arg(m_reply->errorString()), false);
276 } else if (not redirectionTarget.isNull()) {
277 downloadedFile->close();
278 const QUrl newUrl = m_feedURL.resolved(redirectionTarget.toUrl());
279 m_reply->deleteLater();
280 showInformationDialog(tr("Download has started, the installer will open once it's finished downloading"), false);
281 startDownloadFile(newUrl, name);
282 return;
283 } else {
284 setProgress(100); //just in case
285 m_fileSize = 0;
286 downloadedFile->write(m_reply->readAll());
287 downloadedFile->close();
288 auto fileInfo = QFileInfo(*downloadedFile);
289
290#ifdef Q_OS_LINUX
291 QDesktopServices::openUrl(QUrl::fromLocalFile(fileInfo.dir().absolutePath()));
292 QProcess proc;
293 auto res = proc.startDetached(QDir::toNativeSeparators(fileInfo.absoluteFilePath()), QStringList());
294 auto err = proc.error();
295 qDebug() << res << " " << err;
296#else
297
298 QDesktopServices::openUrl(QUrl::fromLocalFile(fileInfo.absoluteFilePath()));
299#endif
300
301 downloadedFile->deleteLater();
302 }

339//---------------------------------------------------------------------------------------------------------------------
340void FvUpdater::httpFeedDownloadFinished() {
341 if (m_httpRequestAborted) {
342 m_reply->deleteLater();
343 return;
344 }
345
346 const QVariant redirectionTarget =
347 m_reply->attribute(QNetworkRequest::RedirectionTargetAttribute);
348 if (m_reply->error() != QNetworkReply::NoError) {
349 // Error.
350 showErrorDialog(
351 tr("Feed download failed: %1.").arg(m_reply->errorString()), false);
352 } else if (not redirectionTarget.isNull()) {
353 const QUrl newUrl = m_feedURL.resolved(redirectionTarget.toUrl());
354
355 m_feedURL = newUrl;
356 m_reply->deleteLater();
357
358 startDownloadFeed(m_feedURL);
359 return;
360 } else {
361 auto jsonDoc = QJsonDocument::fromJson(m_reply->readAll());
362 qDebug() << "Response is a JSON object:" << jsonDoc.isObject();
363 if (jsonDoc.isObject()) {
364 auto tag = jsonDoc.object()["tag_name"].toString();
365 qDebug() << "Found the following tag" << tag;
366
367 if (!releaseIsNewer(tag)) {
368 showInformationDialog(tr("No new releases available."));
369 return;
370 }
371 if (showConfirmationDialog(tr("A new release %1 is available.\nDo you want to download it?").arg(tag), true))
372 getPLatformSpecificInstaller(jsonDoc.object()["assets"].toArray());
373 return;
374 }
375 }
376
377 m_reply->deleteLater();
378
379 if (m_dropOnFinish) {
380 drop();
381 }
382}

384void FvUpdater::getPLatformSpecificInstaller(QJsonArray assets) {
385 qDebug() << "current application version" << QCoreApplication::applicationVersion();
386
387#ifdef Q_OS_LINUX // Defined on Linux.
388 auto searchPattern = "AppImage";
389#else
390 #ifdef Q_OS_MACOS // Defined on macOS
391 auto searchPattern = "macos";
392 #else
393 #ifdef Q_OS_WIN64 // Defined on Windows 64-bit only.
394 auto searchPattern = "windows";
395 #else // Only windows 32-bit left
396 auto searchPattern = "win32";
397 #endif
398 #endif
399#endif
400
401 for (const QJsonValueRef asset : assets) {
402 auto name = asset.toObject()["name"].toString();
403 qDebug() << "Checking" << searchPattern << "against" << name;
404 if (name.contains(searchPattern,
405 Qt::CaseInsensitive)) {
406 QUrl downloadableUrl =
407 asset.toObject()["browser_download_url"].toString();
408 startDownloadFile(downloadableUrl, name);
409 }
410 }
411}

467bool FvUpdater::releaseIsNewer(const QString &releaseTag) {
468 const auto releaseVersion = releaseTag.mid(1).split('.');
469 const auto currentVersion = QCoreApplication::applicationVersion().split('.');
470 for (int i = 0; i < releaseVersion.length(); i++) {
471 if (releaseVersion[i].toInt() > currentVersion[i].toInt())
472 return (m_releaseName = releaseTag), true;
473 }
474 return false;
475}
