71 * @brief DialogCutSpline create dialog.
72 * @param data container with data
73 * @param parent parent widget
74 */
75DialogCutSpline::DialogCutSpline(const VContainer *data, const quint32 &toolId, QWidget *parent)
76 : DialogTool(data, toolId, parent)
77 , ui(new Ui::DialogCutSpline)
78 , formula(QString())
79 , formulaBaseHeight(0)
80{
81 ui->setupUi(this);
82 setWindowFlags(windowFlags() & ~Qt::WindowContextHelpButtonHint);
83 setWindowIcon(QIcon(":/toolicon/32x32/spline_cut_point.png"));
84
85 // Set the position that the dialog opens based on user preference.
86 setDialogPosition();
87
88 ui->lineEditNamePoint->setClearButtonEnabled(true);
89
90 initializeFormulaUi(ui);
91 ui->lineEditNamePoint->setText(qApp->getCurrentDocument()->GenerateLabel(LabelType::NewLabel));
92 labelEditNamePoint = ui->labelEditNamePoint;
93 this->formulaBaseHeight = ui->plainTextEditFormula->height();
94 ui->plainTextEditFormula->installEventFilter(this);
95
96 initializeOkCancelApply(ui);
97 flagFormula = false;
98 DialogTool::CheckState();
99
100 FillComboBoxSplines(ui->comboBoxSpline);
101
102 ui->direction_ComboBox->addItem(tr("Forward (from start point)"), "forward");
103 ui->direction_ComboBox->addItem(tr("Backward (from end point)"), "backward");
104
105 connect(ui->toolButtonExprLength, &QPushButton::clicked, this, &DialogCutSpline::FXLength);
106 connect(ui->lineEditNamePoint, &QLineEdit::textChanged, this, &DialogCutSpline::NamePointChanged);
107 connect(ui->plainTextEditFormula, &QPlainTextEdit::textChanged, this, &DialogCutSpline::FormulaChanged);
108 connect(ui->pushButtonGrowLength, &QPushButton::clicked, this, &DialogCutSpline::DeployFormulaTextEdit);
109
110 vis = new VisToolCutSpline(data);
111}

149 * @param value formula
150 */
151void DialogCutSpline::SetFormula(const QString &value)
152{
153 formula = qApp->translateVariables()->FormulaToUser(value, qApp->Settings()->getOsSeparator());
154 // increase height if needed. TODO : see if I can get the max number of caracters in one line
155 // of this PlainTextEdit to change 80 to this value
156 if (formula.length() > 80)
157 {
158 this->DeployFormulaTextEdit();
159 }
160 ui->plainTextEditFormula->setPlainText(formula);
161
162 VisToolCutSpline *path = qobject_cast<VisToolCutSpline *>(vis);
163 SCASSERT(path != nullptr)
164 path->setLength(formula);
165
166 MoveCursorToEnd(ui->plainTextEditFormula);
167}

189void DialogCutSpline::ChosenObject(quint32 id, const SceneObject &type)
190{

195 if (SetObject(id, ui->comboBoxSpline, ""))

206//---------------------------------------------------------------------------------------------------------------------
207void DialogCutSpline::SaveData()
208{
209 pointName = ui->lineEditNamePoint->text();
210 formula = ui->plainTextEditFormula->toPlainText();
211 formula.replace("\n", " ");
212
213 VisToolCutSpline *path = qobject_cast<VisToolCutSpline *>(vis);
214 SCASSERT(path != nullptr)
215
216 path->setObject1Id(getSplineId());
217 path->setDirection(getDirection());
218 path->setLength(formula);
219 path->RefreshGeometry();
220}
