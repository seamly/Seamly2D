150/*
151 * @brief Handles emitted signal from the scene if an object was selected.
152 * @param selected true is object was selected.
153 * @param objId Object Id of the selected object.
154 * @param toolId Tool Id of the selected object.
155 */
156void InsertNodesDialog::SelectedObject(bool selected, quint32 objId, quint32 toolId)
157{
158 Q_UNUSED(toolId)
159
160 if (prepare)
161 {
162 return;
163 }
164
165 auto nodeIterator = std::find_if(m_nodes.begin(), m_nodes.end(),
166 [objId](const VPieceNode &node) { return node.GetId() == objId; });
167 if (selected)
168 {
169 if (nodeIterator == m_nodes.cend())
170 {
171 GOType objType = GOType::Unknown;
172 try
173 {
174 objType = data->GetGObject(objId)->getType();
175 }
176 catch (const VExceptionBadId &)
177 {
178 qWarning() << "Cannot find an object with id" << objId;
179 return;
180 }
181 m_beep->play();
182 bool appendCurve = false;
183 QSharedPointer<VGObject> previousObj = nullptr;
184 quint32 previousObjId = getLastNodeId();
185 if (previousObjId != NULL_ID)
186 {
187 previousObj = data->GetGObject(previousObjId);
188 }
189
190 VPieceNode node;
191 VPieceNode node2;
192 switch (objType)
193 {
194 case GOType::Point:
195 node = VPieceNode(objId, Tool::NodePoint);
196 if (previousObj != nullptr)
197 {
198 GOType previousObjType = previousObj->getType();
199 if (previousObjType == GOType::Arc || previousObjType == GOType::EllipticalArc ||
200 previousObjType == GOType::Spline || previousObjType == GOType::CubicBezier)
201 {
202 const QSharedPointer<VAbstractCurve> curve = data->GeometricObject<VAbstractCurve>(previousObjId);
203 const QPointF point = static_cast<QPointF>(*data->GeometricObject<VPointF>(objId));
204
205 if (curve->isPointOnCurve(point) &&
206 point != curve->getFirstPoint() &&
207 point != curve->getLastPoint())
208 {
209 switch (previousObjType)
210 {
211 case GOType::Arc:
212 node2 = VPieceNode(previousObjId, Tool::NodeArc, m_nodes.last().GetReverse());
213 appendCurve = true;
214 break;
215 case GOType::EllipticalArc:
216 node2 = VPieceNode(previousObjId, Tool::NodeElArc, m_nodes.last().GetReverse());
217 appendCurve = true;
218 break;
219 case GOType::Spline:
220 case GOType::CubicBezier:
221 node2 = VPieceNode(previousObjId, Tool::NodeSpline, m_nodes.last().GetReverse());
222 appendCurve = true;
223 break;
224 case GOType::SplinePath:
225 case GOType::CubicBezierPath:
226 node2 = VPieceNode(previousObjId, Tool::NodeSplinePath, m_nodes.last().GetReverse());
227 appendCurve = true;
228 break;
229 case GOType::Point:
230 case GOType::Unknown:
231 default:
232 break;
233 }
234 }
235 }
236 }
237 break;
238 case GOType::Arc:
239 node = VPieceNode(objId, Tool::NodeArc, correctCurveDirection(objId));
240 break;
241 case GOType::EllipticalArc:
242 node = VPieceNode(objId, Tool::NodeElArc, correctCurveDirection(objId));
243 break;
244 case GOType::Spline:
245 case GOType::CubicBezier:
246 node = VPieceNode(objId, Tool::NodeSpline, correctCurveDirection(objId));
247 break;
248 case GOType::SplinePath:
249 case GOType::CubicBezierPath:
250 node = VPieceNode(objId, Tool::NodeSplinePath, correctCurveDirection(objId));
251 break;
252 case GOType::Unknown:
253 default:
254 qWarning() << "Ignore unknown object type.";
255 return;
256 }
257
258 node.SetExcluded(true);
259 m_nodes.append(node);
260 if (appendCurve)
261 {
262 node2.SetExcluded(true);
263 m_nodes.append(node2);
264 }
265 if (objType != GOType::Point)
266 {
267 insertCurveNodes(node);
268 }
269 }
270 }
271 else
272 {
273 if (nodeIterator != m_nodes.end())
274 {
275 m_nodes.erase(nodeIterator);
276 }
277 }
278}

281/*
282 * @brief Displays a context menu for the list of node items.
283 * Allows user to set notch type of point nodes.
284 * Allows user to reverse a curve node.
285 * Allows user to delete a node from the list.
286 * @param pos Mouse position on the selected list item to display the menu at.
287 */
288void InsertNodesDialog::showContextMenu(const QPoint &pos)
289{
290 const int row = ui->nodes_ListWidget->currentRow();
291 if (ui->nodes_ListWidget->count() == 0 || row == -1 || row >= ui->nodes_ListWidget->count())
292 {
293 return;
294 }
295
296 // workaround for https://bugreports.qt.io/browse/QTBUG-97559: assign parent to QMenu
297 QScopedPointer<QMenu> menu(new QMenu(ui->nodes_ListWidget));
298 NodeInfo info;
299 NotchType notchType = NotchType::Slit;
300 bool isNotch = false;
301 QListWidgetItem *rowItem = ui->nodes_ListWidget->item(row);
302 SCASSERT(rowItem != nullptr);
303 VPieceNode rowNode = qvariant_cast<VPieceNode>(rowItem->data(Qt::UserRole));
304
305 QAction *actionNotch = nullptr;
306 QAction *actionNone = nullptr;
307 QAction *actionSlit = nullptr;
308 QAction *actionTNotch = nullptr;
309 QAction *actionUNotch = nullptr;
310 QAction *actionVInternal = nullptr;
311 QAction *actionVExternal = nullptr;
312 QAction *actionCastle = nullptr;
313 QAction *actionDiamond = nullptr;
314 QAction *actionReverse = nullptr;
315
316 if (rowNode.GetTypeTool() != Tool::NodePoint)
317 {
318 actionReverse = menu->addAction(tr("Reverse"));
319 actionReverse->setCheckable(true);
320 actionReverse->setChecked(rowNode.GetReverse());
321 }
322 else
323 {
324
325 QMenu *notchMenu = menu->addMenu(tr("Notch"));
326 actionNotch = notchMenu->menuAction();
327 actionNotch->setCheckable(true);
328 actionNotch->setChecked(rowNode.isNotch());
329
330 actionNone = notchMenu->addAction( tr("None"));
331 actionSlit = notchMenu->addAction(QIcon("://icon/24x24/slit_notch.png"), tr("Slit"));
332 actionTNotch = notchMenu->addAction(QIcon("://icon/24x24/t_notch.png"), tr("TNotch"));
333 actionUNotch = notchMenu->addAction(QIcon("://icon/24x24/u_notch.png"), tr("UNotch"));
334 actionVInternal = notchMenu->addAction(QIcon("://icon/24x24/internal_v_notch.png"), tr("VInternal"));
335 actionVExternal = notchMenu->addAction(QIcon("://icon/24x24/external_v_notch.png"), tr("VExternal"));
336 actionCastle = notchMenu->addAction(QIcon("://icon/24x24/castle_notch.png"), tr("Castle"));
337 actionDiamond = notchMenu->addAction(QIcon("://icon/24x24/diamond_notch.png"), tr("Diamond"));
338 }
339
340 QAction *actionDelete = menu->addAction(QIcon::fromTheme("edit-delete"), tr("Delete") + QStringLiteral("\tDel"));
341 QAction *selectedAction = menu->exec(ui->nodes_ListWidget->viewport()->mapToGlobal(pos));
342 if (selectedAction == actionReverse && rowNode.GetTypeTool() != Tool::NodePoint)
343 {
344 rowNode.SetReverse(!rowNode.GetReverse());
345 info = getNodeInfo(rowNode, true);
346 rowItem->setData(Qt::UserRole, QVariant::fromValue(rowNode));
347 rowItem->setIcon(QIcon(info.icon));
348 rowItem->setText(info.name);
349 }
350 else if (selectedAction == actionDelete)
351 {
352 delete rowItem;
353 if (ui->nodes_ListWidget->count() == 0)
354 {
355 ui->statusMsg_Label->setText(tr("No nodes selected. Press Cancel to continue"));
356 }
357 }
358 else
359 {
360 if (selectedAction == actionNone)
361 {
362 isNotch = false;
363 notchType = NotchType::Slit;
364 }
365 else if (selectedAction == actionSlit)
366 {
367 isNotch = true;
368 notchType = NotchType::Slit;
369 }
370 else if (selectedAction == actionTNotch)
371 {
372 isNotch = true;
373 notchType = NotchType::TNotch;
374 }
375 else if (selectedAction == actionUNotch)
376 {
377 isNotch = true;
378 notchType = NotchType::UNotch;
379 }
380 else if (selectedAction == actionVInternal)
381 {
382 isNotch = true;
383 notchType = NotchType::VInternal;
384 }
385 else if (selectedAction == actionVExternal)
386 {
387 isNotch = true;
388 notchType = NotchType::VExternal;
389 }
390 else if (selectedAction == actionCastle)
391 {
392 isNotch = true;
393 notchType = NotchType::Castle;
394 }
395 else if (selectedAction == actionDiamond)
396 {
397 isNotch = true;
398 notchType = NotchType::Diamond;
399 }
400
401 rowNode.setNotch(isNotch);
402 rowNode.setNotchType(notchType);
403 info = getNodeInfo(rowNode, true);
404 rowItem->setData(Qt::UserRole, QVariant::fromValue(rowNode));
405 rowItem->setIcon(QIcon(info.icon));
406 rowItem->setText(info.name);
407 }
408
409 validateNodes();
410}

424/*
425 * @brief Filters keyboard event to check if the delete key was pressed.
426 * @param object QObject that sent the event.
427 * @param event QEvent.
428 * @return If the node list was the sending object: True if the key pressed was the Delete key False if any other key.
429 * If event sent by any other object pass the event on to the parent.
430 */
431bool InsertNodesDialog::eventFilter(QObject *object, QEvent *event)
432{
433 if (QListWidget *list = qobject_cast<QListWidget *>(object))
434 {
435 if (event->type() == QEvent::KeyPress)
436 {
437 QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
438 if (keyEvent->key() == Qt::Key_Delete)
439 {
440 const int row = ui->nodes_ListWidget->currentRow();
441 if (ui->nodes_ListWidget->count() > 0 || row > -1 || row < ui->nodes_ListWidget->count())
442 {
443 QListWidgetItem *rowItem = list->item(row);
444 SCASSERT(rowItem != nullptr);
445 delete rowItem;
446 }
447 return true;
448 }
449 }
450 }
451 else
452 {
453 return DialogTool::eventFilter(object, event); // pass the event on to the parent class
454 }
455 return false; // pass the event to the widget
456}

502/*
503 * @brief Checks whether a selected curve needs to be reversed so it's direction is clockwise.
504 * @param curveObjId Id of selected node.
505 * @return true if curve needs to be reversed or false if curve does not need to be reversed.
506*/
507bool InsertNodesDialog::correctCurveDirection(quint32 curveObjId)
508{
509 const QSharedPointer<VGObject> curveObj = data->GetGObject(curveObjId);
510 const QString curveName = curveObj->name();
511 quint32 nodeId;
512 const QSharedPointer<VAbstractCurve> curve = data->GeometricObject<VAbstractCurve>(curveObjId);
513
514 for (auto &node : m_nodes)
515 {
516 nodeId = node.GetId();
517 const QSharedPointer<VGObject> nodeObj = data->GetGObject(nodeId);
518 if (static_cast<GOType>(nodeObj->getType()) == GOType::Point)
519 {
520 const QPointF point = static_cast<QPointF>(*data->GeometricObject<VPointF>(nodeId));
521 if (point == curve->getLastPoint())
522 {
523 ui->statusMsg_Label->setText(ui->statusMsg_Label->text() + curveName +
524 tr(" was auto reversed.") + "\n");
525 return true;
526 }
527 else if (point == curve->getFirstPoint())
528 {
529 return false;
530 }
531 }
532 }
533 ui->statusMsg_Label->setText(ui->statusMsg_Label->text() + curveName +
534 tr(" may need to be manually reversed.") + "\n");
535 return false;
536}

539/*
540 * @brief Checks whether to insert of copy of the selected curve before any node points that may exist on the curve.
541 * @param curveNode selected curve node.
542 */
543void InsertNodesDialog::insertCurveNodes(VPieceNode curveNode)
544{
545 quint32 nodeId;
546 quint32 prevNodeId;
547 quint32 curveNodeId = curveNode.GetId();
548
549 const QSharedPointer<VAbstractCurve> curve = data->GeometricObject<VAbstractCurve>(curveNodeId);
550 for (int i = 0; i < m_nodes.size(); ++i)
551 {
552 nodeId = m_nodes.value(i).GetId();
553 const QSharedPointer<VGObject> nodeObj = data->GetGObject(nodeId);
554 if (static_cast<GOType>(nodeObj->getType()) == GOType::Point)
555 {
556 const QPointF point = static_cast<QPointF>(*data->GeometricObject<VPointF>(nodeId));
557 if (curve->isPointOnCurve(point) &&
558 point != curve->getFirstPoint() &&
559 point != curve->getLastPoint() &&
560 curveNodeId != nodeId)
561 {
562 if (i > 0)
563 {
564 prevNodeId = m_nodes.value(i - 1).GetId();
565 }
566 if (i == 0 || prevNodeId != curveNodeId)
567 {
568 VPieceNode node2 = curveNode;
569 node2.SetExcluded(true);
570 m_nodes.insert(i, 1, node2);
571 }
572 }
573 }
