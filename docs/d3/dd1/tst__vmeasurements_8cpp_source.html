72/**
73 * @brief CreateEmptyMultisizeFile check if empty multisize measurement file is valid.
74 */
75void TST_Measurements::CreateEmptyMultisizeFile()
76{
77 Unit mUnit = Unit::Cm;
78 const int height = 176;
79 const int size = 50;
80
81 QSharedPointer<VContainer> data = QSharedPointer<VContainer>(new VContainer(nullptr, &mUnit));
82 VContainer::setHeight(height);
83 VContainer::setSize(size);
84
85 QSharedPointer<MeasurementDoc> m =
86 QSharedPointer<MeasurementDoc>(new MeasurementDoc(mUnit, size, height, data.data()));
87 m->setSize(VContainer::rsize());
88 m->setHeight(VContainer::rheight());
89
90 QTemporaryFile file;
91 QString fileName;
92 // In Windows we have problems when we try to open QSaveFile when QTemporaryFile with the same name is already open.
93 if (file.open())
94 {
95 // So, before we try to open file in m->SaveDocument function we need to close it and remove.
96 // Just closing - is not enough, if we just close QTemporaryFile we get "access denied" in Windows.
97 fileName = file.fileName();
98 file.close();
99 file.remove();
100 QString error;
101 const bool result = m->SaveDocument(fileName, error);
102
103 QVERIFY2(result, error.toUtf8().constData());
104 }
105 else
106 {
107 QFAIL("Can't open temporary file.");
108 }
109
110 try
111 {
112 VDomDocument::ValidateXML(MultiSizeConverter::CurrentSchema, fileName);
113 }
114 catch (VException &error)
115 {
116 QFAIL(error.ErrorMessage().toUtf8().constData());
117 }
118}

161/**
162 * @brief ValidPMCodesMultisizeFile helps to check that all current pattern making systems match pattern inside XSD
163 * scheme.
164 */
165void TST_Measurements::ValidPMCodesMultisizeFile()
166{
167 Unit mUnit = Unit::Cm;
168 const int height = 176;
169 const int size = 50;
170
171 QSharedPointer<VContainer> data = QSharedPointer<VContainer>(new VContainer(nullptr, &mUnit));
172 VContainer::setHeight(height);
173 VContainer::setSize(size);
174
175 QSharedPointer<MeasurementDoc> m =
176 QSharedPointer<MeasurementDoc>(new MeasurementDoc(mUnit, size, height, data.data()));
177 m->setSize(VContainer::rsize());
178 m->setHeight(VContainer::rheight());
179
180 const QStringList listSystems = ListPMSystems();
181 for (int i = 0; i < listSystems.size(); ++i)
182 {
183 QString code = listSystems.at(i);
184 code.remove(0, 1); // remove 'p'
185 m->SetPMSystem(code);
186
187 QTemporaryFile file;
188 QString fileName;
189 if (file.open())
190 {
191 fileName = file.fileName();
192 file.close();
193 file.remove();
194 QString error;
195 const bool result = m->SaveDocument(fileName, error);
196
197 const QString message = QString("Error: %1 for code=%2").arg(error).arg(listSystems.at(i));
198 QVERIFY2(result, qUtf8Printable(message));
199 }
200 else
201 {
202 QFAIL("Can't open temporary file.");
203 }
204
205 try
206 {
207 VDomDocument::ValidateXML(MultiSizeConverter::CurrentSchema, fileName);
208 }
209 catch (VException &error)
210 {
211 const QString message = QString("Error: %1 for code=%2").arg(error.ErrorMessage()).arg(listSystems.at(i));
212 QFAIL(qUtf8Printable(message));
213 }
214 }
215}
