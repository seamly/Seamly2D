259/**
260 * @brief ChosenObject gets id and type of selected object. Save right data and ignore wrong.
261 * @param id id of point or detail
262 * @param type type of object
263 */
264void PointIntersectXYDialog::ChosenObject(quint32 id, const SceneObject &type)
265{
266 if (prepare == false)// After first choose we ignore all objects
267 {
268 if (type == SceneObject::Point)
269 {
270 PointIntersectXYVisual *visual = qobject_cast<PointIntersectXYVisual *>(vis);
271 SCASSERT(visual != nullptr)
272
273 switch (number)
274 {
275 case 0:
276 if (SetObject(id, ui->firstPoint_ComboBox, tr("Select point for Y value (horizontal)")))
277 {
278 number++;
279 visual->setPoint1Id(id);
280 visual->RefreshGeometry();
281 }
282 break;
283 case 1:
284 if (getCurrentObjectId(ui->firstPoint_ComboBox) != id)
285 {
286 if (SetObject(id, ui->secondPoint_ComboBox, ""))
287 {
288 visual->setPoint2Id(id);
289 visual->RefreshGeometry();
290 prepare = true;
291 DialogAccepted();
292 }
293 }
294 break;
295 default:
296 break;
297 }
298 }
299 }
300}

303void PointIntersectXYDialog::pointChanged()
304{
305 QColor color = okColor;
306 if (getFirstPointId() == getSecondPointId())
307 {
308 flagError = false;
309 color = errorColor;
310 }
311 else
312 {
313 flagError = true;
314 color = okColor;
315 }
316 ChangeColor(ui->firstPoint_Label, color);
317 ChangeColor(ui->secondPoint_Label, color);
318 CheckState();
319}
