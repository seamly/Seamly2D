161//---------------------------------------------------------------------------------------------------------------------
162void DialogPointOfIntersectionArcs::ChosenObject(quint32 id, const SceneObject &type)
163{
164 if (prepare == false)// After first choose we ignore all objects
165 {
166 if (type == SceneObject::Arc)
167 {
168 VisToolPointOfIntersectionArcs *point = qobject_cast<VisToolPointOfIntersectionArcs *>(vis);
169 SCASSERT(point != nullptr)
170
171 switch (number)
172 {
173 case 0:
174 if (SetObject(id, ui->comboBoxArc1, tr("Select second an arc")))
175 {
176 number++;
177 point->VisualMode(id);
178 }
179 break;
180 case 1:
181 if (getCurrentObjectId(ui->comboBoxArc1) != id)
182 {
183 if (SetObject(id, ui->comboBoxArc2, ""))
184 {
185 number = 0;
186 point->setArc2Id(id);
187 point->RefreshGeometry();
188 prepare = true;
189 DialogAccepted();
190 }
191 }
192 break;
193 default:
194 break;
195 }
196 }
197 }
198}
