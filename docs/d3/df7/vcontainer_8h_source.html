90/**
91 * @class VContainerData
92 * @brief A data container class that uses Qt's implicit sharing mechanism.
93 *
94 * The VContainerData class is designed to efficiently manage shared data
95 * using Qt's implicit sharing (copy-on-write) mechanism. It contains
96 * several hash maps and shared pointers to handle various elements.
97 *
98 * @details
99 * The class inherits from QSharedData to leverage Qt's implicit sharing.
100 * It has two constructors: one for initializing its member variables and
101 * another for copying the contents from another instance. The class manages
102 * the following members:
103 *
104 * - `gObjects`: A hash map mapping a quint32 to a shared pointer of VGObject.
105 * - `variables`: A hash map mapping a QString to a shared pointer of VInternalVariable.
106 * - `pieces`: A shared pointer to a hash map mapping a quint32 to VPiece.
107 * - `piecePaths`: A shared pointer to a hash map mapping a quint32 to VPiecePath.
108 * - `trVars`: A pointer to a VTranslateVars object.
109 * - `patternUnit`: A pointer to a Unit object.
110 *
111 * The class provides a constructor for initializing these members and a
112 * copy constructor to create copies of the data container.
113 */
114

117public:

381/**
382 * @brief Retrieves a shared pointer to a geometric object of type T with the specified ID.
383 *
384 * This template method attempts to find and return a geometric object of type T stored in the container
385 * with the given ID. It performs a dynamic cast to ensure the object is of the expected type.
386 *
387 * @tparam T The expected type of the geometric object to retrieve.
388 * @param id The ID of the geometric object to retrieve.
389 * @return const QSharedPointer<T> A shared pointer to the geometric object of type T.
390 *
391 * @throws VExceptionBadId if the object ID is NULL_ID, if the object cannot be found, or if the object cannot be cast to the expected type.
392 *
393 * @details
394 * - The method first checks if the provided ID is `NULL_ID` and throws a `VExceptionBadId` exception if it is.
395 * - It then attempts to find the geometric object with the specified ID in the container.
396 * - If the object is found, it attempts to cast it to the expected type T using `qSharedPointerDynamicCast`.
397 * - If the cast is successful, it returns the shared pointer to the geometric object.
398 * - If the cast fails due to a bad allocation, it throws a `VExceptionBadId` exception with an appropriate message.
399 * - If the object with the specified ID does not exist, it throws a `VExceptionBadId` exception.
400 */
401template <typename T>
402const QSharedPointer<T> VContainer::GeometricObject(const quint32 &id) const
403{
404
405 if (id == NULL_ID)
406 {
407 throw VExceptionBadId(tr("Can't find object"), id);
408 }
409
410 QSharedPointer<VGObject> gObj = QSharedPointer<VGObject>();
411 if (d->gObjects.contains(id))
412 {
413 gObj = d->gObjects.value(id);
414 }
415 else
416 {
417 throw VExceptionBadId(tr("Can't find object Id: "), id);
418 }
419 try
420 {
421 QSharedPointer<T> obj = qSharedPointerDynamicCast<T>(gObj);
422 SCASSERT(obj.isNull() == false)
423 return obj;
424 }
425 catch (const std::bad_alloc &)
426 {
427 throw VExceptionBadId(tr("Can't cast object"), id);
428 }
429}

431/**

433 *
434 * This template method attempts to find and return a variable of type T stored in the container
435 * with the given name. It performs a dynamic cast to ensure the variable is of the expected type.
436 *
437 * @tparam T The expected type of the variable to retrieve.
438 * @param name The name of the variable to retrieve.
439 * @return QSharedPointer<T> A shared pointer to the variable of type T.
440 *
441 * @throws VExceptionBadId if the variable cannot be found or cannot be cast to the expected type.
442 *
443 * @note The method includes assertions to check that the name is not empty and that the cast is successful.
444 *
445 * @details
446 * - The method first asserts that the provided name is not empty.
447 * - It checks if the variable with the specified name exists in the container.
448 * - If the variable exists, it attempts to cast it to the expected type T using `qSharedPointerDynamicCast`.
449 * - If the cast is successful, it returns the shared pointer to the variable.
450 * - If the cast fails due to a bad allocation, it throws a `VExceptionBadId` exception with an appropriate message.

459 try

467 throw VExceptionBadId(tr("Can't cast object"), name);

469 }
470 else
471 {
472 throw VExceptionBadId(tr("Can't find object"), name);
473 }

496/**
497 * @brief Adds or updates a variable with the specified name and shared pointer to the container.
498 *
499 * This template method adds a variable of type T to the container using a shared pointer.
500 * If a variable with the same name already exists and is of the same type, it updates the existing variable.
501 * If the types do not match, it throws an exception.
502 *
503 * @tparam T The type of the variable to add.
504 * @param name The name of the variable to add.
505 * @param var A shared pointer to the variable to add.
506 *
507 * @throws VExceptionBadId if the variable cannot be cast to the expected type or if there is a type mismatch.
508 *
509 * @details
510 * - The method first checks if a variable with the specified name already exists in the container.
511 * - If it exists and is of the same type, it updates the existing variable with the new value.
512 * - If the existing variable cannot be cast to the expected type, it throws a `VExceptionBadId` exception.
513 * - If the types do not match, it throws a `VExceptionBadId` exception.
514 * - If the variable does not already exist, it adds the new variable to the container.
515 * - The method also adds the variable name to a set of unique names.
516 */
517template <typename T>
518void VContainer::AddVariable(const QString& name, const QSharedPointer<T> &var)
519{
520 if (d->variables.contains(name))
521 {
522 if (d->variables.value(name)->GetType() == var->GetType())
523 {
524 QSharedPointer<T> v = qSharedPointerDynamicCast<T>(d->variables.value(name));
525 if (v.isNull())
526 {
527 throw VExceptionBadId(tr("Can't cast object."), name);
528 }
529 *v = *var;
530 }
531 else
532 {
533 throw VExceptionBadId(tr("Can't find object. Type mismatch."), name);
534 }
535 }
536 else
537 {
538 d->variables.insert(name, var);
539 }
540
541 uniqueNames.insert(name);
542}

594/**
595 * @brief Updates a geometric object with the specified ID using a shared pointer.
596 *
597 * This template method updates a geometric object of type T in the container using a shared pointer.
598 * It ensures the shared pointer is not null and delegates to another method to perform the update.
599 *
600 * @tparam T The type of the geometric object to update.
601 * @param id The ID of the geometric object to update.
602 * @param obj A shared pointer to the geometric object to update.
603 *
604 * @details
605 * - The method asserts that the shared pointer is not null.
606 * - It calls the UpdateObject method to update the geometric object in the container with the specified ID.
607 * - It inserts the name of the object into a set of unique names.
608 */
609template <typename T>
610void VContainer::UpdateObject(const quint32 &id, const QSharedPointer<T> &point)
611{
612 Q_ASSERT_X(id != NULL_ID, Q_FUNC_INFO, "id == 0"); //-V654 //-V712
613 SCASSERT(point.isNull() == false)
614 point->setId(id);
