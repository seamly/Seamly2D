242//---------------------------------------------------------------------------------------------------------------------
243void DialogTrueDarts::ChosenObject(quint32 id, const SceneObject &type)
244{
245 if (prepare == false)// After first choose we ignore all objects
246 {
247 if (type == SceneObject::Point)
248 {
249 VisToolTrueDarts *points = qobject_cast<VisToolTrueDarts *>(vis);
250 SCASSERT(points != nullptr)
251
252 switch (number)
253 {
254 case 0:
255 if (SetObject(id, ui->comboBoxFirstBasePoint, tr("Select the second base point")))
256 {
257 number++;
258 points->VisualMode(id);
259 }
260 break;
261 case 1:
262 if (getCurrentObjectId(ui->comboBoxFirstBasePoint) != id)
263 {
264 if (SetObject(id, ui->comboBoxSecondBasePoint, tr("Select the first dart point")))
265 {
266 number++;
267 points->setObject2Id(id);
268 points->RefreshGeometry();
269 }
270 }
271 break;
272 case 2:
273 {
274 QSet<quint32> set;
275 set.insert(getCurrentObjectId(ui->comboBoxFirstBasePoint));
276 set.insert(getCurrentObjectId(ui->comboBoxSecondBasePoint));
277 set.insert(id);
278
279 if (set.size() == 3)
280 {
281 if (SetObject(id, ui->comboBoxFirstDartPoint, tr("Select the second dart point")))
282 {
283 number++;
284 points->setD1PointId(id);
285 points->RefreshGeometry();
286 }
287 }
288 break;
289 }
290 case 3:
291 {
292 QSet<quint32> set;
293 set.insert(getCurrentObjectId(ui->comboBoxFirstBasePoint));
294 set.insert(getCurrentObjectId(ui->comboBoxSecondBasePoint));
295 set.insert(getCurrentObjectId(ui->comboBoxFirstDartPoint));
296 set.insert(id);
297
298 if (set.size() == 4)
299 {
300 if (SetObject(id, ui->comboBoxSecondDartPoint, tr("Select the third dart point")))
301 {
302 number++;
303 points->setD2PointId(id);
304 points->RefreshGeometry();
305 }
306 }
307 break;
308 }
309 case 4:
310 {
311 QSet<quint32> set;
312 set.insert(getCurrentObjectId(ui->comboBoxFirstBasePoint));
313 set.insert(getCurrentObjectId(ui->comboBoxSecondBasePoint));
314 set.insert(getCurrentObjectId(ui->comboBoxFirstDartPoint));
315 set.insert(getCurrentObjectId(ui->comboBoxSecondDartPoint));
316 set.insert(id);
317
318 if (set.size() == 5)
319 {
320 if (SetObject(id, ui->comboBoxThirdDartPoint, ""))
321 {
322 points->setD3PointId(id);
323 points->RefreshGeometry();
324 prepare = true;
325 DialogAccepted();
326 }
327 }
328 break;
329 }
330 default:
331 break;
332 }
333 }
334 }

338void DialogTrueDarts::PointNameChanged()
339{
340 QSet<quint32> set;
341 set.insert(getCurrentObjectId(ui->comboBoxFirstBasePoint));
342 set.insert(getCurrentObjectId(ui->comboBoxSecondBasePoint));
343 set.insert(getCurrentObjectId(ui->comboBoxFirstDartPoint));
344 set.insert(getCurrentObjectId(ui->comboBoxSecondDartPoint));
345 set.insert(getCurrentObjectId(ui->comboBoxThirdDartPoint));
346
347 QColor color = okColor;

353 else

355 flagError = true;

438void DialogTrueDarts::CheckName(QLineEdit *edit, QLabel *labelEditNamePoint, const QString &pointD1Name,
439 const QString &pointD2Name, QLineEdit *secondPointName, bool &flagName)
440{
441 SCASSERT(labelEditNamePoint != nullptr)
442 SCASSERT(secondPointName != nullptr)
443 SCASSERT(edit != nullptr)
444
445 const QString name = edit->text();
446 const QString secondName = secondPointName->text();
447 QRegularExpression rx(NameRegExp());
448 if (name.isEmpty()
449 || secondName == name
450 || (pointD1Name != name && pointD2Name != name && data->IsUnique(name) == false)
451 || rx.match(name).hasMatch() == false)
452 {
453 flagName = false;
454 ChangeColor(labelEditNamePoint, Qt::red);
455 }
456 else
457 {
458 flagName = true;
459 ChangeColor(labelEditNamePoint, okColor);
460 }
