200void DialogCubicBezier::ChosenObject(quint32 id, const SceneObject &type)
201{
202 if (prepare == false)// After first choose we ignore all objects
203 {
204 if (type == SceneObject::Point)
205 {
206 auto *path = qobject_cast<VisToolCubicBezier *>(vis);
207 SCASSERT(path != nullptr)
208
209 switch (number)
210 {
211 case 0:
212 if (SetObject(id, ui->comboBoxP1, tr("Select the second point of curve")))
213 {
214 ++number;
215 path->VisualMode(id);
216 }
217 break;
218 case 1:
219 if (SetObject(id, ui->comboBoxP2, tr("Select the third point of curve")))
220 {
221 ++number;
222
223 path->setObject2Id(id);
224 path->RefreshGeometry();
225 }
226 break;
227 case 2:
228 if (SetObject(id, ui->comboBoxP3, tr("Select the fourth point of curve")))
229 {
230 ++number;
231
232 path->setObject3Id(id);
233 path->RefreshGeometry();
234 }
235 break;
236 case 3:
237 if (getCurrentObjectId(ui->comboBoxP1) != id)
238 {
239 if (SetObject(id, ui->comboBoxP4, ""))
240 {
241 ++number;
242
243 path->setObject4Id(id);
244 path->RefreshGeometry();
245 prepare = true;
246 DialogAccepted();
247 }
248 }
249 break;
250 default:
251 break;
252 }
253 }
254 }
255}

257//---------------------------------------------------------------------------------------------------------------------
258void DialogCubicBezier::PointNameChanged()
259{
260 QColor color = okColor;
261 if (getCurrentObjectId(ui->comboBoxP1) == getCurrentObjectId(ui->comboBoxP4))
262 {
263 flagError = false;
264 color = errorColor;
265
266 ui->lineEditSplineName->setText(tr("Invalid spline"));
267 }
268 else
269 {
270 flagError = true;
271 color = okColor;
272
273 if (getCurrentObjectId(ui->comboBoxP1) == spl.GetP1().id() &&
274 getCurrentObjectId(ui->comboBoxP4) == spl.GetP4().id())
275 {
276 newDuplicate = -1;
277 ui->lineEditSplineName->setText(qApp->translateVariables()->VarToUser(spl.name()));
278 }
279 else
280 {
281 VCubicBezier spline(*GetP1(), *GetP2(), *GetP3(), *GetP4());
282
283 if (not data->IsUnique(spline.name()))
284 {
285 newDuplicate = static_cast<qint32>(DNumber(spline.name()));
286 spline.SetDuplicate(static_cast<quint32>(newDuplicate));
287 }
288 ui->lineEditSplineName->setText(qApp->translateVariables()->VarToUser(spline.name()));
289 }
290 }
291 ChangeColor(ui->labelName, color);
292 ChangeColor(ui->labelFirstPoint, color);
293 ChangeColor(ui->labelSecondPoint, color);
294 ChangeColor(ui->labelThirdPoint, color);
295 ChangeColor(ui->labelForthPoint, color);
296 CheckState();
297}

306void DialogCubicBezier::SaveData()
307{
308 const auto p1 = GetP1();
309 const auto p2 = GetP2();
310 const auto p3 = GetP3();
311 const auto p4 = GetP4();
312
313 spl = VCubicBezier(*p1, *p2, *p3, *p4);
314
315 const quint32 d = spl.GetDuplicate();//Save previous value
316 newDuplicate <= -1 ? spl.SetDuplicate(d) : spl.SetDuplicate(static_cast<quint32>(newDuplicate));
317
318 auto path = qobject_cast<VisToolCubicBezier *>(vis);
319 SCASSERT(path != nullptr)
320
321 path->setObject1Id(p1->id());
322 path->setObject2Id(p2->id());
323 path->setObject3Id(p3->id());
324 path->setObject4Id(p4->id());
325 path->SetMode(Mode::Show);
326 path->RefreshGeometry();
327}
