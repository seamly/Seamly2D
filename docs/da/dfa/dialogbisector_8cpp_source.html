210/**
211 * @brief ChosenObject gets id and type of selected object. Save right data and ignore wrong.
212 * @param id id of point or detail
213 * @param type type of object
214 */
215void DialogBisector::ChosenObject(quint32 id, const SceneObject &type)
216{
217 if (prepare == false)// After first choose we ignore all objects
218 {
219 if (type == SceneObject::Point)
220 {
221 VisToolBisector *line = qobject_cast<VisToolBisector *>(vis);
222 SCASSERT(line != nullptr)
223
224 switch (number)
225 {
226 case 0:
227 if (SetObject(id, ui->comboBoxFirstPoint, tr("Select second point of angle")))
228 {
229 number++;
230 line->VisualMode(id);
231 }
232 break;
233 case 1:
234 if (getCurrentObjectId(ui->comboBoxFirstPoint) != id)
235 {
236 if (SetObject(id, ui->comboBoxSecondPoint, tr("Select third point of angle")))
237 {
238 number++;
239 line->setObject2Id(id);
240 line->RefreshGeometry();
241 }
242 }
243 break;
244 case 2:
245 {
246 QSet<quint32> set;
247 set.insert(getCurrentObjectId(ui->comboBoxFirstPoint));
248 set.insert(getCurrentObjectId(ui->comboBoxSecondPoint));
249 set.insert(id);
250
251 if (set.size() == 3)
252 {
253 if (SetObject(id, ui->comboBoxThirdPoint, ""))
254 {
255 line->setObject3Id(id);
256 line->RefreshGeometry();
257 prepare = true;
258 this->setModal(true);
259 this->show();
260 }
261 }
262 }
263 break;
264 default:
265 break;
266 }
267 }
268 }
269}

408void DialogBisector::SaveData()
409{
410 pointName = ui->lineEditNamePoint->text();
411
412 formula = ui->plainTextEditFormula->toPlainText();
413 formula.replace("\n", " ");
414
415 VisToolBisector *line = qobject_cast<VisToolBisector *>(vis);
416 SCASSERT(line != nullptr)
417
418 line->setObject1Id(GetFirstPointId());
419 line->setObject2Id(GetSecondPointId());
420 line->setObject3Id(GetThirdPointId());
421 line->setLength(formula);
