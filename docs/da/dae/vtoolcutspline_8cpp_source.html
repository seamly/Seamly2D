148/**
149 * @brief Create help create tool.
150 * @param _id tool id, 0 if tool doesn't exist yet.
151 * @param pointName point name.
152 * @param formula string with formula length first spline.
153 * @param splineId id spline in data container.
154 * @param mx label bias x axis.
155 * @param my label bias y axis.
156 * @param scene pointer to scene.
157 * @param doc dom document container.
158 * @param data container with variables.
159 * @param parse parser file mode.
160 * @param typeCreation way we create this tool.
161 */
162VToolCutSpline* VToolCutSpline::Create(const quint32 _id, const QString &pointName, QString &direction,
163 QString &formula, const quint32 &splineId, qreal mx, qreal my,
164 bool showPointName, VMainGraphicsScene *scene, VAbstractPattern *doc,
165 VContainer *data, const Document &parse, const Source &typeCreation)
166{
167 const auto spl = data->GeometricObject<VAbstractCubicBezier>(splineId);
168 SCASSERT(splPath != nullptr)
169
170 qreal splLength = qApp->fromPixel(spl->GetLength());
171 QString adjFormula = formula;
172 if (direction == "backward")
173 {
174 adjFormula = QString("%1 - %2").arg(splLength).arg(formula);
175 }
176 const qreal length = CheckFormula(_id, adjFormula, data);
177
178 QPointF spl1p2, spl1p3, spl2p2, spl2p3;
179 QPointF point = spl->CutSpline(qApp->toPixel(length), spl1p2, spl1p3, spl2p2, spl2p3);
180
181 quint32 id = _id;
182 VPointF *p = new VPointF(point, pointName, mx, my);
183 p->setShowPointName(showPointName);
184
185 auto spline1 = QSharedPointer<VAbstractBezier>(new VSpline(spl->GetP1(), spl1p2, spl1p3, *p));
186 auto spline2 = QSharedPointer<VAbstractBezier>(new VSpline(*p, spl2p2, spl2p3, spl->GetP4()));
187
188 if (typeCreation == Source::FromGui)
189 {
190 id = data->AddGObject(p);
191 data->AddSpline(spline1, NULL_ID, id);
192 data->AddSpline(spline2, NULL_ID, id);
193 }
194 else
195 {
196 data->UpdateGObject(id, p);
197 data->AddSpline(spline1, NULL_ID, id);
198 data->AddSpline(spline2, NULL_ID, id);
199
200 if (parse != Document::FullParse)
201 {
202 doc->UpdateToolData(id, data);
203 }
204 }
205
206 if (parse == Document::FullParse)
207 {
208 VDrawTool::AddRecord(id, Tool::CutSpline, doc);
209 VToolCutSpline *point = new VToolCutSpline(doc, data, id, direction, formula, splineId, typeCreation);
210 scene->addItem(point);
211 InitToolConnections(scene, point);
212 VAbstractPattern::AddTool(id, point);
213 doc->IncrementReferens(spl->getIdTool());
214 return point;
215 }
216 return nullptr;
217}

297QString VToolCutSpline::makeToolTip() const
298{
299 const auto spl = VAbstractTool::data.GeometricObject<VAbstractCubicBezier>(curveCutId);
300
301 const QString expression = qApp->translateVariables()->FormulaToUser(formula, qApp->Settings()->getOsSeparator());
302 const qreal length = Visualization::FindVal(expression, VAbstractTool::data.DataVariables());
303
304 QPointF spl1p2, spl1p3, spl2p2, spl2p3;
305 QPointF point = spl->CutSpline(qApp->toPixel(length), spl1p2, spl1p3, spl2p2, spl2p3);
306
307 VSpline spline1 = VSpline(spl->GetP1(), spl1p2, spl1p3, VPointF(point));
308 VSpline spline2 = VSpline(VPointF(point), spl2p2, spl2p3, spl->GetP4());
309
310 const QString curveStr = tr("Curve");
311 const QString lengthStr = tr("length");
312
313 const QString toolTip = QString("<table style=font-size:11pt; font-weight:600>"
314 "<tr> <td><b>%6:</b> %7</td> </tr>"
315 "<tr> <td><b>%1:</b> %2 %3</td> </tr>"
316 "<tr> <td><b>%8:</b> %9</td> </tr>"
317 "<tr> <td><b>%4:</b> %5 %3</td> </tr>"
318 "</table>")
319 .arg(curveStr + QLatin1String("1 ") + lengthStr)
320 .arg(qApp->fromPixel(spline1.GetLength()))
321 .arg(UnitsToStr(qApp->patternUnit(), true))
322 .arg(curveStr + QLatin1String("2 ") + lengthStr)
323 .arg(qApp->fromPixel(spline2.GetLength()))
324 .arg(curveStr + QLatin1String(" 1") + tr("label"))
325 .arg(spline1.name())
326 .arg(curveStr + QLatin1String(" 2") + tr("label"))
327 .arg(spline2.name());
328
329 return toolTip;
