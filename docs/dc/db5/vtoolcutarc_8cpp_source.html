144//---------------------------------------------------------------------------------------------------------------------
145/**
146 * @brief Create help create tool.
147 * @param _id tool id, 0 if tool doesn't exist yet.
148 * @param pointName point name.
149 * @param formula string with formula length first arc.
150 * @param arcId id arc in data container.
151 * @param mx label bias x axis.
152 * @param my label bias y axis.
153 * @param scene pointer to scene.
154 * @param doc dom document container.
155 * @param data container with variables.
156 * @param parse parser file mode.
157 * @param typeCreation way we create this tool.
158 */
159VToolCutArc* VToolCutArc::Create(const quint32 _id, const QString &pointName, QString &direction,
160 QString &formula, quint32 arcId, qreal mx, qreal my, bool showPointName,
161 VMainGraphicsScene *scene, VAbstractPattern *doc,
162 VContainer *data, const Document &parse, const Source &typeCreation)
163{
164 const QSharedPointer<VArc> arc = data->GeometricObject<VArc>(arcId);
165 SCASSERT(arc != nullptr)
166
167 qreal arcLength = qApp->fromPixel(arc->GetLength());
168 QString adjFormula = formula;
169 if (direction == "backward")
170 {
171 adjFormula = QString("%1 - %2").arg(arcLength).arg(formula);
172 }
173 const qreal length = CheckFormula(_id, adjFormula, data);
174
175 VArc arc1;
176 VArc arc2;
177 QPointF point = arc->CutArc(qApp->toPixel(length), arc1, arc2);
178
179 quint32 id = _id;
180 VPointF *p = new VPointF(point, pointName, mx, my);
181 p->setShowPointName(showPointName);
182
183 auto a1 = QSharedPointer<VArc>(new VArc(arc1));
184 auto a2 = QSharedPointer<VArc>(new VArc(arc2));
185 if (typeCreation == Source::FromGui)
186 {
187 id = data->AddGObject(p);
188 a1->setId(VContainer::getNextId());
189 a2->setId(VContainer::getNextId());
190 data->AddArc(a1, a1->id(), id);
191 data->AddArc(a2, a2->id(), id);
192 }
193 else
194 {
195 data->UpdateGObject(id, p);
196 a1->setId(id + 1);
197 a2->setId(id + 2);
198 data->AddArc(a1, a1->id(), id);
199 data->AddArc(a2, a2->id(), id);
200
201 if (parse != Document::FullParse)
202 {
203 doc->UpdateToolData(id, data);
204 }
205 }
206
207 if (parse == Document::FullParse)
208 {
209 VDrawTool::AddRecord(id, Tool::CutArc, doc);
210 VToolCutArc *point = new VToolCutArc(doc, data, id, direction, formula, arcId, typeCreation);
211 scene->addItem(point);
212 InitToolConnections(scene, point);
213 VAbstractPattern::AddTool(id, point);
214 doc->IncrementReferens(arc->getIdTool());
215 return point;
216 }
217 return nullptr;
218}

298QString VToolCutArc::makeToolTip() const
299{
300 const QSharedPointer<VArc> arc = VAbstractTool::data.GeometricObject<VArc>(curveCutId);
301
302 const QString expression = qApp->translateVariables()->FormulaToUser(formula, qApp->Settings()->getOsSeparator());
303 const qreal length = Visualization::FindVal(expression, VAbstractTool::data.DataVariables());
304
305 const QString arcStr = tr("Arc");
306 const QString lengthStr = tr("length");
307 const QString startAngleStr = tr("start angle");
308 const QString endAngleStr = tr("end angle");
309 const QString radiusStr = tr("radius");
310
311 VArc ar1;
312 VArc ar2;
313 arc->CutArc(qApp->toPixel(length), ar1, ar2);
314
315 auto ArcToolTip = [arcStr, lengthStr, startAngleStr, endAngleStr, radiusStr](QString toolTip, const VArc &arc,
316 const QString &arcNumber)
317 {
318 toolTip += QString("<table style=font-size:11pt; font-weight:600>"
319 "<tr> <td><b>%10:</b> %11</td> </tr>"
320 "<tr> <td><b>%1:</b> %2 %3</td> </tr>"
321 "<tr> <td><b>%4:</b> %5 %3</td> </tr>"
322 "<tr> <td><b>%6:</b> %7</td> </tr>"
323 "<tr> <td><b>%8:</b> %9</td> </tr>"
324 "</table>")
325 .arg(arcStr + arcNumber + QLatin1String(" ") + lengthStr)
326 .arg(qApp->fromPixel(arc.GetLength()))
327 .arg(UnitsToStr(qApp->patternUnit(), true))
328 .arg(arcStr + arcNumber + QLatin1String(" ") + radiusStr)
329 .arg(qApp->fromPixel(arc.GetRadius()))
330 .arg(arcStr + arcNumber + QLatin1String(" ") + startAngleStr)
331 .arg(arc.GetStartAngle())
332 .arg(arcStr + arcNumber + QLatin1String(" ") + endAngleStr)
333 .arg(arc.GetEndAngle())
334 .arg(arcStr + arcNumber + QLatin1String(" ") + tr("label"))
335 .arg(arc.name());
336 return toolTip;
337 };
338
339 return ArcToolTip(ArcToolTip("<table>", ar1, "1"), ar2, "2") + QLatin1String("</table>");
