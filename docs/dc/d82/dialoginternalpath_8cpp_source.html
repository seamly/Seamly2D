119void DialogInternalPath::ChosenObject(quint32 id, const SceneObject &type)
120{
121 if (not prepare)
122 {
123 bool reverse = false;
124 if (QGuiApplication::keyboardModifiers() == Qt::ShiftModifier)
125 {
126 reverse = true;
127 }
128 if (id != GetLastId())
129 {
130 switch (type)
131 {
132 case SceneObject::Arc:
133 NewItem(VPieceNode(id, Tool::NodeArc, reverse));
134 break;
135 case SceneObject::ElArc:
136 NewItem(VPieceNode(id, Tool::NodeElArc, reverse));
137 break;
138 case SceneObject::Point:
139 NewItem(VPieceNode(id, Tool::NodePoint));
140 break;
141 case SceneObject::Spline:
142 NewItem(VPieceNode(id, Tool::NodeSpline, reverse));
143 break;
144 case SceneObject::SplinePath:
145 NewItem(VPieceNode(id, Tool::NodeSplinePath, reverse));
146 break;
147 case (SceneObject::Line):
148 case (SceneObject::Piece):
149 case (SceneObject::Unknown):
150 default:
151 qWarning() << "Got wrong scene object. Ignore.";
152 break;
153 }
154 }
155 else
156 {
157 if (ui->listWidget->count() > 1)
158 {
159 delete getItemById(id);
160 }
161 }
162
163 ValidObjects(PathIsValid());
164
165 if (not m_showMode)
166 {
167 auto visPath = qobject_cast<VisToolInternalPath *>(vis);
168 SCASSERT(visPath != nullptr);
169 const VPiecePath p = CreatePath();
170 visPath->SetPath(p);
171
172 if (p.CountNodes() == 1)
173 {
174 emit ToolTip(tr("Select main path objects, Use <b>SHIFT</b> to reverse curve direction, "
175 "Press <b>ENTER</b> to finish path creation "));
176
177 if (not qApp->getCurrentScene()->items().contains(visPath))
178 {
179 visPath->VisualMode(NULL_ID);
180 }
181 else
182 {
183 visPath->RefreshGeometry();
184 }
185 }
186 else
187 {
188 visPath->RefreshGeometry();
189 }
190 }
191 }
192}
