37SeamlyMeWelcomeDialog::SeamlyMeWelcomeDialog(QWidget *parent)
38 : QDialog(parent)
39 , ui(new Ui::SeamlyMeWelcomeDialog)
40 , m_langChanged(false)
41 , settings(qApp->seamlyMeSettings())
42{

45
46 //-------------------- Units setup

51 ui->separator_CheckBox->setChecked(settings->getOsSeparator());

53
54 //-------------------- Languages setup

56 connect(ui->language_ComboBox, static_cast<void (QComboBox::*)(int)>(&QComboBox::currentIndexChanged),

64 QPushButton *ok_Button = ui->buttonBox->button(QDialogButtonBox::Ok);

76// @brief apply apply dialog changes
77void SeamlyMeWelcomeDialog::apply()
78{
79 settings->SetUnit(qvariant_cast<QString>(ui->units_ComboBox->currentData()));
80 settings->setOsSeparator(ui->separator_CheckBox->isChecked());
81 settings->getOsSeparator() ? setLocale(QLocale()) : setLocale(QLocale::c());
82 settings->setShowWelcome(ui->doNotShow_CheckBox->isChecked());
83
84 if (m_langChanged)
85 {
86 const QString locale = qvariant_cast<QString>(ui->language_ComboBox->currentData());
87 settings->setLocale(locale);
88 m_langChanged = false;
89 }
90
91 done(QDialog::Accepted);
92}
