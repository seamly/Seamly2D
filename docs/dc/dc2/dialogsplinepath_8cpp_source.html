264/**
265 * @brief ChosenObject gets id and type of selected object. Save right data and ignore wrong.
266 * @param id id of point or detail
267 * @param type don't show this id in list
268 */
269void DialogSplinePath::ChosenObject(quint32 id, const SceneObject &type)
270{
271 if (type == SceneObject::Point)
272 {
273 if (AllIds().contains(id))
274 {
275 return;
276 }
277
278 const auto point = data->GeometricObject<VPointF>(id);
279 VSplinePoint p;
280 p.SetP(*point);
281 NewItem(p);
282 emit ToolTip(tr("Select point of curve path"));
283
284 SavePath();
285
286 auto visPath = qobject_cast<VisToolSplinePath *>(vis);
287 SCASSERT(visPath != nullptr)
288 visPath->setPath(path);
289
290 if (path.CountPoints() == 1)
291 {
292 visPath->VisualMode(NULL_ID);
293 VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
294 SCASSERT(window != nullptr)
295 connect(visPath, &VisToolSplinePath::ToolTip, window, &VAbstractMainWindow::ShowToolTip);
296
297 connect(visPath, &VisToolSplinePath::PathChanged, this, &DialogSplinePath::PathUpdated);
298 }
299 else
300 {
301 visPath->RefreshGeometry();
302 }
303 }
304}

324void DialogSplinePath::CheckState()
325{
326 SCASSERT(ok_Button != nullptr)
327
328 bool fAngle1 = true, fAngle2 = true, fLength1 = true, fLength2 = true;
329
330 for (qint32 i = 0; i < ui->listWidget->count(); ++i)
331 {
332 fAngle1 = fAngle1 && flagAngle1.at(i);
333 fAngle2 = fAngle2 && flagAngle2.at(i);
334 fLength1 = fLength1 && flagLength1.at(i);
335 fLength2 = fLength2 && flagLength2.at(i);
336 }
337
338 ok_Button->setEnabled(fAngle1 && fAngle2 && fLength1 && fLength2 && flagError);
339 // In case dialog does not have an apply button
340 if (apply_Button != nullptr)
341 {
342 apply_Button->setEnabled(ok_Button->isEnabled());
343 }
344}

392//---------------------------------------------------------------------------------------------------------------------
393void DialogSplinePath::Angle1Changed()
394{
395 const int row = ui->listWidget->currentRow();
396 if (row == -1)
397 {
398 return;
399 }
400
401 if (row != 0)
402 {
403 QListWidgetItem *item = ui->listWidget->item(row);
404 SCASSERT(item != nullptr)
405 VSplinePoint p = qvariant_cast<VSplinePoint>(item->data(Qt::UserRole));
406
407 const QString angle1F = ui->plainTextEditAngle1F->toPlainText().replace("\n", " ");
408 const qreal angle1 = Visualization::FindVal(angle1F, data->DataVariables());
409
410 try
411 {
412 p.SetAngle1(angle1, qApp->translateVariables()->FormulaFromUser(angle1F, qApp->Settings()->getOsSeparator()));
413 }
414 catch (qmu::QmuParserError &error)
415 {
416 Q_UNUSED(error)
417 p.SetAngle1(angle1, angle1F);
418 }
419
420 item->setData(Qt::UserRole, QVariant::fromValue(p));
421
422 EvalAngle1();
423
424 if (row != ui->listWidget->count()-1)
425 {
426 ui->plainTextEditAngle2F->blockSignals(true);
427 ui->plainTextEditAngle2F->setPlainText(qApp->translateVariables()->FormulaToUser(p.Angle2Formula(),
428 qApp->Settings()->getOsSeparator()));
429 EvalAngle2();
430 ui->plainTextEditAngle2F->blockSignals(false);
431 }
432 }
433}
