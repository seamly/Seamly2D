183//---------------------------------------------------------------------------------------------------------------------
184void DialogPointOfIntersectionCurves::ChosenObject(quint32 id, const SceneObject &type)
185{
186 if (prepare == false)// After first choose we ignore all objects
187 {
188 if (type == SceneObject::Spline
189 || type == SceneObject::Arc
190 || type == SceneObject::ElArc
191 || type == SceneObject::SplinePath)
192 {
193 auto point = qobject_cast<VisToolPointOfIntersectionCurves *>(vis);
194 SCASSERT(point != nullptr)
195
196 switch (number)
197 {
198 case 0:
199 if (SetObject(id, ui->comboBoxCurve1, tr("Select second curve")))
200 {
201 number++;
202 point->VisualMode(id);
203 }
204 break;
205 case 1:
206 if (getCurrentObjectId(ui->comboBoxCurve1) != id)
207 {
208 if (SetObject(id, ui->comboBoxCurve2, ""))
209 {
210 number = 0;
211 point->setObject2Id(id);
212 point->RefreshGeometry();
213 prepare = true;
214 DialogAccepted();
215 }
216 }
217 break;
218 default:
219 break;
220 }
221 }
222 }
223}
