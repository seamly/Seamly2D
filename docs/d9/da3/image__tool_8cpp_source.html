48QString getImageFilename(QWidget *parent)
49{
50 const QString filter = QObject::tr("Images") + QLatin1String(" (*.bmp *.gif *.jpeg *.jpg *.pbm *.pgm *.png *.ppm *.svg *.tif *.xbm *.xpm);;") +
51 "BMP" + QLatin1String(" (*.bmp);;") +
52 "GIF" + QLatin1String(" (*.gif);;") +
53 "JPEG" + QLatin1String(" (*.jpeg);;") +
54 "JPG" + QLatin1String(" (*.jpg);;") +
55 "PBM" + QLatin1String(" (*.pbm);;") +
56 "PGM" + QLatin1String(" (*.pgm);;") +
57 "PNG" + QLatin1String(" (*.png);;") +
58 "PPM" + QLatin1String(" (*.ppm);;") +
59 "SVG" + QLatin1String(" (*.svg);;") +
60 "TIF" + QLatin1String(" (*.tif);;") +
61 "XBM" + QLatin1String(" (*.xbm);;") +
62 "XPM" + QLatin1String(" (*.xpm)");
63
64 const QString path = qApp->Seamly2DSettings()->getImageFilePath();
65
66 bool usedNotExistedDir = false;
67 QDir directory(path);
68 if (!directory.exists())
69 {
70 usedNotExistedDir = directory.mkpath(".");
71 }
72
73 const QString filename = QFileDialog::getOpenFileName(parent, QObject::tr("Open Image File"), path, filter, nullptr,
74 QFileDialog::DontUseNativeDialog);
75
76 return filename;
77}

81ImageTool::ImageTool(QObject *parent, VAbstractPattern *doc, VMainGraphicsScene *draftScene, QString filename)
82 : QObject(parent),
83 m_doc(doc),
84 m_draftScene(draftScene)
85{
86 qCDebug(vImageTool, "Image filename: %s", qUtf8Printable(filename));
87
88 if (filename.isEmpty())
89 {
90 qCDebug(vImageTool, "Can't load image");
91 QMessageBox::critical(nullptr, tr("Import Image"), tr("Could not load the image."), QMessageBox::Ok);
92 return;
93 }
94
95 QFileInfo f(filename);
96
97 image.name = f.baseName();
98 image.filename = filename;
99 image.units = qApp->patternUnit();
100
101 image.id = VContainer::getNextId();
102
103 addImage();
104}

106//Constructor called when file is full-parsed
107ImageTool::ImageTool(QObject *parent, VAbstractPattern *doc, VMainGraphicsScene *draftScene, DraftImage img)
108 : QObject(parent),
109 image(img),
110 m_doc(doc),
111 m_draftScene(draftScene)
112{
113 qCDebug(vImageTool, "Image filename: %s", qUtf8Printable(image.filename));
114
115 QFileInfo fileInfo(image.filename);
116
117 if (!fileInfo.exists())
118 {
119 const QString text = tr("The image <br/><br/> <b>%1</b> <br/><br/> could not be found. Do you "
120 "want to update the file location?").arg(image.filename);
121 QMessageBox::StandardButton result = QMessageBox::question(nullptr, tr("Loading image"), text,
122 QMessageBox::Yes | QMessageBox::No,
123 QMessageBox::Yes);
124 if (result == QMessageBox::No)
125 {
126 return;
127 }
128 else
129 {
130 bool askAgain = true;
131 while(askAgain)
132 {
133 image.filename = getImageFilename(nullptr);
134 if (image.filename.isEmpty())
135 {
136 qCDebug(vImageTool, "No image selected");
137 QMessageBox::critical(nullptr, tr("Import Image"), tr("No image was selected..."), QMessageBox::Ok);
138 continue;
139 }
140
141 QImageReader imageReader(image.filename);
142
143 if(!imageReader.canRead())
144 {
145 qCDebug(vImageTool, "Can't read image");
146 QMessageBox::critical(nullptr, tr("Import Image"), tr("Could not read the image.") + "\n" + tr("File may be corrupted..."), QMessageBox::Ok);
147 continue;
148 }
149 askAgain = false;
150 }
151 image.name = fileInfo.baseName();
152
153 QDomElement domElement = m_doc->elementById(image.id, VAbstractPattern::TagDraftImage);
154 QDomElement newDomElement = domElement.cloneNode().toElement();
155 m_doc->SetAttribute(newDomElement, VAbstractPattern::AttrName, image.name);
156 m_doc->SetAttribute(newDomElement, VAbstractPattern::AttrSource, image.filename);
157 domElement.parentNode().replaceChild(newDomElement, domElement);
158 }
159 }
160
161 if (image.name.isEmpty())
162 {
163 image.name = fileInfo.baseName();
164 }
165
166 addImage();
167}

170void ImageTool::addImage()
171{
172 QImageReader imageReader(image.filename);
173
174 if(!imageReader.canRead())
175 {
176 qCDebug(vImageTool, "Can't read image");
177 QMessageBox::critical(nullptr, tr("Import Image"), tr("Could not read the image.") + "\n" + tr("File may be corrupted..."), QMessageBox::Ok);
178 return;
179 }
180
181 imageItem = new ImageItem(this, m_doc, image);
182 m_doc->addBackgroundImage(image.id, imageItem);
183 m_draftScene->addItem(imageItem);
184 //Need error dialog
185
186 //connect(this, &MainWindow::EnableImageSelection, item, &ImageItem::enableSelection);

189 //connect(item, &ImageItem::imageSelected, this, &MainWindow::handleImageSelected);
190 connect(imageItem, &ImageItem::setStatusMessage, this, [this](QString message) {emit setStatusMessage(message);});
191 connect(imageItem, &ImageItem::imageNeedsSave, this, &ImageTool::saveChanges);
192
193 creationWasSuccessful = true;
194}

198void ImageTool::handleDeleteImage(quint32 id)
199{
200 Q_UNUSED(id);
201
202 qCDebug(vImageTool, "delete Image");
203 bool deleteConfirmed = true;
204
205 if (qApp->Settings()->getConfirmItemDelete())
206 {
207 Utils::CheckableMessageBox msgBox(qApp->getMainWindow());
208 msgBox.setWindowTitle(tr("Confirm deletion"));
209 msgBox.setText(tr("Do you really want to delete?"));
210 msgBox.setStandardButtons(QDialogButtonBox::Yes | QDialogButtonBox::No);
211 msgBox.setDefaultButton(QDialogButtonBox::No);
212 msgBox.setIconPixmap(QApplication::style()->standardIcon(QStyle::SP_MessageBoxQuestion).pixmap(32, 32) );
213
214 int dialogResult = msgBox.exec();
215
216 if (dialogResult != QDialog::Accepted)
217 {
218 deleteConfirmed = false;
219 }
220
221 qApp->Settings()->setConfirmItemDelete(not msgBox.isChecked());
222
223 }
224
225 if (deleteConfirmed)
226 {
227 imageItem->deleteImageItem();
228
229 DelTool *delTool = new DelTool(m_doc, id);
230 connect(delTool, &DelTool::NeedFullParsing, m_doc, &VAbstractPattern::NeedFullParsing);
231 qApp->getUndoStack()->push(delTool);
232 }
233}
