99QString MultiSizeConverter::getSchema(int ver) const
100{
101 switch (ver)
102 {
103 case (0x000300):
104 return QStringLiteral("://schema/multi_size_measurements/v0.3.0.xsd");
105 case (0x000400):
106 return QStringLiteral("://schema/multi_size_measurements/v0.4.0.xsd");
107 case (0x000401):
108 return QStringLiteral("://schema/multi_size_measurements/v0.4.1.xsd");
109 case (0x000402):
110 return QStringLiteral("://schema/multi_size_measurements/v0.4.2.xsd");
111 case (0x000403):
112 return QStringLiteral("://schema/multi_size_measurements/v0.4.3.xsd");
113 case (0x000404):
114 return QStringLiteral("://schema/multi_size_measurements/v0.4.4.xsd");
115 case (0x000405):
116 return CurrentSchema;
117 default:
118 InvalidVersion(ver);
119 break;
120 }
121 return QString();//unreachable code
122}

234void MultiSizeConverter::convertMeasurementsToV0_4_0()
235{
236 // TODO. Delete if minimal supported version is 0.4.0
237 Q_STATIC_ASSERT_X(MultiSizeConverter::MeasurementMinVer < CONVERTER_VERSION_CHECK(0, 4, 0),
238 "Time to refactor the code.");
239
240 const QString tagBM = QStringLiteral("body-measurements");
241
242 QDomElement bm = createElement(tagBM);
243
244 const QMultiMap<QString, QString> names = OldNamesToNewNames_InV0_3_0();
245 const QList<QString> keys = names.uniqueKeys();
246 for (int i = 0; i < keys.size(); ++i)
247 {
248 qreal resValue = 0;
249 qreal resSizeIncrease = 0;
250 qreal resHeightIncrease = 0;
251
252 // This has the same effect as a .values(), just isn't as elegant
253 const QList<QString> list = names.values( keys.at(i) );
254 foreach(const QString &val, list )
255 {
256 const QDomNodeList nodeList = this->elementsByTagName(val);
257 if (nodeList.isEmpty())
258 {
259 continue;
260 }
261
262 QDomElement m = nodeList.at(0).toElement();
263 const qreal value = GetParametrDouble(m, QStringLiteral("value"), "0.0");
264 const qreal size_increase = GetParametrDouble(m, QStringLiteral("size_increase"), "0.0");
265 const qreal height_increase = GetParametrDouble(m, QStringLiteral("height_increase"), "0.0");
266
267 if (not qFuzzyIsNull(value))
268 {
269 resValue = value;
270 resSizeIncrease = size_increase;
271 resHeightIncrease = height_increase;
272 }
273 }
274
275 bm.appendChild(addMeasurementsVer0_4_0(keys.at(i), resValue, resSizeIncrease, resHeightIncrease));
276 }
277
278 QDomElement rootElement = this->documentElement();
279 const QDomNodeList listBM = elementsByTagName(tagBM);
280 rootElement.replaceChild(bm, listBM.at(0));
281}
