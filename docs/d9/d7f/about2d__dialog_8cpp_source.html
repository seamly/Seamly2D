204void About2DAppDialog::setProgressValue(int val) {
205 if (!ui->downloadProgress->isVisible()){
206 ui->downloadProgress->show();
207 ui->pushButtonCheckUpdate->setDisabled(true);
208 }
209 ui->downloadProgress->setValue(val);
210 if (val == 100){
211 ui->downloadProgress->hide();
212 ui->downloadProgress->setValue(0);
213 ui->pushButtonCheckUpdate->setDisabled(false);
214 }
215}
