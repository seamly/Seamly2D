162void SeamlyMePreferencesConfigurationPage::Apply()
163{
164 VSeamlyMeSettings *settings = qApp->seamlyMeSettings();
165
166 settings->setShowWelcome(ui->showWelcome_CheckBox->isChecked());
167 settings->setOsSeparator(ui->osOption_CheckBox->isChecked());
168 settings->setToolBarStyle(ui->toolBarStyle_CheckBox->isChecked());
169
170 if (m_langChanged || m_systemChanged)
171 {
172 const QString locale = qvariant_cast<QString>(ui->langCombo->currentData());
173 settings->setLocale(locale);
174 m_langChanged = false;
175
176 const QString code = qvariant_cast<QString>(ui->systemCombo->currentData());
177 settings->SetPMSystemCode(code);
178 m_systemChanged = false;
179
180 qApp->loadTranslations(locale);
181 qApp->processEvents();// force to call changeEvent
182
183 // Part about measurments will not be updated automatically
184 qApp->retranslateTables();
185 qApp->retranslateGroups();
186 }
187
188 if (m_defGradationChanged)
189 {
190 settings->SetDefHeight(ui->defHeightCombo->currentText().toInt());
191 settings->SetDefSize(ui->defSizeCombo->currentText().toInt());
192 m_defGradationChanged = false;
193 }
194}
