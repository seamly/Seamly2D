82//---------------------------------------------------------------------------------------------------------------------
83void VisToolLineIntersect::RefreshGeometry()
84{
85 if (object1Id > NULL_ID)
86 {
87 const QSharedPointer<VPointF> first = Visualization::data->GeometricObject<VPointF>(object1Id);
88 DrawPoint(line1P1, static_cast<QPointF>(*first), supportColor);
89
90 if (line1P2Id <= NULL_ID)
91 {
92 DrawLine(line1, QLineF(static_cast<QPointF>(*first), Visualization::scenePos), supportColor, lineWeight);
93 }
94 else
95 {
96 const QSharedPointer<VPointF> second = Visualization::data->GeometricObject<VPointF>(line1P2Id);
97 DrawPoint(line1P2, static_cast<QPointF>(*second), supportColor);
98
99 DrawLine(line1, QLineF(static_cast<QPointF>(*first), static_cast<QPointF>(*second)),
100 supportColor, lineWeight);
101
102 if (line2P1Id <= NULL_ID)
103 {
104 return;
105 }
106 else
107 {
108 const QSharedPointer<VPointF> third = Visualization::data->GeometricObject<VPointF>(line2P1Id);
109 DrawPoint(line2P1, static_cast<QPointF>(*third), supportColor);
110
111 if (line2P2Id <= NULL_ID)
112 {
113 DrawLine(this, QLineF(static_cast<QPointF>(*third), Visualization::scenePos),
114 supportColor, lineWeight);
115
116 QLineF l1(static_cast<QPointF>(*first), static_cast<QPointF>(*second));
117 QLineF l2(static_cast<QPointF>(*third), Visualization::scenePos);
118 QPointF fPoint;
119 QLineF::IntersectType intersect = l1.intersects(l2, &fPoint);
120 if (intersect == QLineF::UnboundedIntersection || intersect == QLineF::BoundedIntersection)
121 {
122 DrawPoint(point, fPoint, mainColor);
123 }
124 }
125 else
126 {
127 const QSharedPointer<VPointF> forth = Visualization::data->GeometricObject<VPointF>(line2P2Id);
128 DrawPoint(line2P2, static_cast<QPointF>(*forth), supportColor);
129
130 DrawLine(this, QLineF(static_cast<QPointF>(*third), static_cast<QPointF>(*forth)),
131 supportColor, lineWeight);
132
133 QLineF l1(static_cast<QPointF>(*first), static_cast<QPointF>(*second));
134 QLineF l2(static_cast<QPointF>(*third), static_cast<QPointF>(*forth));
135 QPointF fPoint;
136 QLineF::IntersectType intersect = l1.intersects(l2, &fPoint);
137 if (intersect == QLineF::UnboundedIntersection || intersect == QLineF::BoundedIntersection)
138 {
139 DrawPoint(point, fPoint, mainColor);
140 }
141 }
142 }
143 }
144 }
145}
