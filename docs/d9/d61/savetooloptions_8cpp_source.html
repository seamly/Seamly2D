75void SaveToolOptions::undo()
76{

78
79 QDomElement domElement = doc->elementById(nodeId);
80 if (domElement.isElement())

113bool SaveToolOptions::mergeWith(const QUndoCommand *command)
114{
115 const SaveToolOptions *saveCommand = static_cast<const SaveToolOptions *>(command);
116 SCASSERT(saveCommand != nullptr)
117 const quint32 id = saveCommand->getToolId();
118
119 if (id != nodeId)
120 {
121 return false;
122 }
123
124 newXml = saveCommand->getNewXml();
125 return true;
126}
