146void MeShortcutsDialog::exportPdf()
147{
148 QString filename = QFileDialog::getSaveFileName(nullptr, tr("Export PDF"), QString(),
149 "*.pdf", nullptr, QFileDialog::DontUseNativeDialog);
150
151 if (QFileInfo(filename).suffix().isEmpty())
152 {
153 filename.append(".pdf");
154 }
155 QPrinter printer(QPrinter::PrinterResolution);
156 printer.setOutputFormat(QPrinter::PdfFormat);
157 printer.setPageSize(printer.pageLayout().pageSize());
158 printer.setOutputFileName(filename);
159
160 QTextDocument textDocument;
161 textDocument.setHtml(ui->shortcuts_TextBrowser->toHtml());
162 textDocument.setPageSize(printer.pageLayout().paintRectPixels(printer.resolution()).size());
163 textDocument.print(&printer);
164}
