72DialogLine::DialogLine(const VContainer *data, const quint32 &toolId, QWidget *parent)
73 : DialogTool(data, toolId, parent)
74 , ui(new Ui::DialogLine)
75{
76 ui->setupUi(this);
77 setWindowFlags(windowFlags() & ~Qt::WindowContextHelpButtonHint);
78 setWindowIcon(QIcon(":/toolicon/32x32/line.png"));
79
80 // Set the position that the dialog opens based on user preference.
81 setDialogPosition();
82
83 initializeOkCancelApply(ui);
84
85 fillComboBoxPoints(ui->comboBoxFirstPoint);
86 fillComboBoxPoints(ui->comboBoxSecondPoint);
87
88 int index = ui->lineType_ComboBox->findData(LineTypeNone);
89 if (index != -1)
90 {
91 ui->lineType_ComboBox->removeItem(index);
92 }
93
94 index = ui->lineColor_ComboBox->findData(qApp->getCurrentDocument()->getDefaultLineColor());
95 if (index != -1)
96 {
97 ui->lineColor_ComboBox->setCurrentIndex(index);
98 }
99
100 index = ui->lineWeight_ComboBox->findData(qApp->getCurrentDocument()->getDefaultLineWeight());
101 if (index != -1)
102 {
103 ui->lineWeight_ComboBox->setCurrentIndex(index);
104 }
105
106 index = ui->lineType_ComboBox->findData(qApp->getCurrentDocument()->getDefaultLineType());
107 if (index != -1)
108 {
109 ui->lineType_ComboBox->setCurrentIndex(index);
110 }
111
112 number = 0;
113
114 connect(ui->comboBoxFirstPoint, &QComboBox::currentTextChanged, this, &DialogLine::PointNameChanged);
115 connect(ui->comboBoxSecondPoint, &QComboBox::currentTextChanged, this, &DialogLine::PointNameChanged);
116
117 vis = new VisToolLine(data);
118}

262/**
263 * @brief ChosenObject gets id and type of selected object. Save right data and ignore wrong.
264 * @param id id of point or detail
265 * @param type type of object
266 */
267void DialogLine::ChosenObject(quint32 id, const SceneObject &type)
268{
269 if (prepare == false)// After first choose we ignore all objects
270 {
271 if (type == SceneObject::Point)
272 {
273 switch (number)
274 {
275 case 0:
276 if (SetObject(id, ui->comboBoxFirstPoint, tr("Select second point")))
277 {
278 number++;
279 vis->VisualMode(id);
280 }
281 break;
282 case 1:
283 if (getCurrentObjectId(ui->comboBoxFirstPoint) != id)
284 {
285 if (SetObject(id, ui->comboBoxSecondPoint, ""))
286 {
287 if (flagError)
288 {
289 number = 0;
290 prepare = true;
291 DialogAccepted();
292 }
293 }
294 }
295 break;
296 default:
297 break;
298 }
299 }
300 }
301}
