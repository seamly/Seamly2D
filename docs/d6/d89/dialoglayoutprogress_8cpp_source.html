108void DialogLayoutProgress::Error(const LayoutErrors &state)
109{
110 switch (state)
111 {
112 case LayoutErrors::NoError:
113 return;
114 case LayoutErrors::PrepareLayoutError:
115 qCritical() << tr("Couldn't prepare data for creation layout");
116 break;
117 case LayoutErrors::EmptyPaperError:
118 qCritical() << tr("One or more pattern pieces are bigger than the paper format you selected. Please select a bigger paper format.");
119 break;
120 case LayoutErrors::ProcessStoped:
121 default:
122 break;
123 }
124
125 done(QDialog::Rejected);
126}
