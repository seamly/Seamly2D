112/**
113 * @brief ChosenObject gets id and type of selected object. Save right data and ignore wrong.
114 * @param id id of point or detail
115 * @param type type of object
116 */
117void DialogTriangle::ChosenObject(quint32 id, const SceneObject &type)
118{
119 if (prepare == false)// After first choose we ignore all objects
120 {
121 if (type == SceneObject::Point)
122 {
123 VisToolTriangle *line = qobject_cast<VisToolTriangle *>(vis);
124 SCASSERT(line != nullptr)
125
126 switch (number)
127 {
128 case (0):
129 if (SetObject(id, ui->comboBoxAxisP1, tr("Select second point of axis")))
130 {
131 number++;
132 line->VisualMode(id);
133 }
134 break;
135 case (1):
136 if (getCurrentObjectId(ui->comboBoxAxisP1) != id)
137 {
138 if (SetObject(id, ui->comboBoxAxisP2, tr("Select first point")))
139 {
140 number++;
141 line->setObject2Id(id);
142 line->RefreshGeometry();
143 }
144 }
145 break;
146 case (2):
147 {
148 QSet<quint32> set;
149 set.insert(getCurrentObjectId(ui->comboBoxAxisP1));
150 set.insert(getCurrentObjectId(ui->comboBoxAxisP2));
151 set.insert(id);
152
153 if (set.size() == 3)
154 {
155 if (SetObject(id, ui->comboBoxFirstPoint, tr("Select second point")))
156 {
157 number++;
158 line->setHypotenuseP1Id(id);
159 line->RefreshGeometry();
160 }
161 }
162 }
163 break;
164 case (3):
165 {
166 QSet<quint32> set;
167 set.insert(getCurrentObjectId(ui->comboBoxAxisP1));
168 set.insert(getCurrentObjectId(ui->comboBoxAxisP2));
169 set.insert(getCurrentObjectId(ui->comboBoxFirstPoint));
170 set.insert(id);
171
172 if (set.size() == 4)
173 {
174 if (SetObject(id, ui->comboBoxSecondPoint, ""))
175 {
176 line->setHypotenuseP2Id(id);
177 line->RefreshGeometry();
178 prepare = true;
179 DialogAccepted();
180 }
181 }
182 }
183 break;
184 default:
185 break;
186 }
187 }
188 }
189}

192void DialogTriangle::SaveData()
193{
194 pointName = ui->lineEditNamePoint->text();
195
196 VisToolTriangle *line = qobject_cast<VisToolTriangle *>(vis);
197 SCASSERT(line != nullptr)
198
199 line->setObject1Id(GetAxisP1Id());
200 line->setObject2Id(GetAxisP2Id());
201 line->setHypotenuseP1Id(GetFirstPointId());
202 line->setHypotenuseP2Id(GetSecondPointId());
203 line->RefreshGeometry();
204}

206//---------------------------------------------------------------------------------------------------------------------
207void DialogTriangle::PointNameChanged()
208{
209 QSet<quint32> set;
210 set.insert(getCurrentObjectId(ui->comboBoxFirstPoint));
211 set.insert(getCurrentObjectId(ui->comboBoxSecondPoint));
212 set.insert(getCurrentObjectId(ui->comboBoxAxisP1));
213 set.insert(getCurrentObjectId(ui->comboBoxAxisP2));
214
215 QColor color = okColor;
216 if (set.size() < 3)//Need tree or more unique points for creation triangle
217 {
218 flagError = false;
219 color = errorColor;
220 }
221 else
222 {
223 flagError = true;
224 color = okColor;
225 }
226 ChangeColor(ui->labelFirstPoint, color);
227 ChangeColor(ui->labelSecondPoint, color);
228 ChangeColor(ui->labelAxisP1, color);
229 ChangeColor(ui->labelAxisP2, color);
230 CheckState();
231}
