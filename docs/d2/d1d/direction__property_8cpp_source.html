53QVariant VPE::DirectionProperty::data(int column, int role) const
54{
55 if (m_directions.empty())
56 {
57 return QVariant();
58 }
59
60 int tmpIndex = VProperty::d_ptr->VariantValue.toInt();
61
62 if (tmpIndex < 0 || tmpIndex >= m_indexList.count())
63 {
64 tmpIndex = 0;
65 }
66
67 if (column == DPC_Data && Qt::DisplayRole == role)
68 {
69 return m_indexList.at(tmpIndex);
70 }
71 else if (column == DPC_Data && Qt::EditRole == role)
72 {
73 return tmpIndex;
74 }
75 else
76 {
77 return VProperty::data(column, role);
78 }
79}

81QWidget *VPE::DirectionProperty::createEditor(QWidget *parent, const QStyleOptionViewItem &options,
82 const QAbstractItemDelegate *delegate)
83{
84 Q_UNUSED(options)
85 Q_UNUSED(delegate)
86 QComboBox* directionEditor = new QComboBox(parent);
87
88 directionEditor->clear();
89 directionEditor->setMinimumWidth(140);
90 directionEditor->setLocale(parent->locale());
91
92 QMap<QString, QString>::const_iterator i = m_directions.constBegin();
93 while (i != m_directions.constEnd())
94 {
95 directionEditor->addItem(i.value(), QVariant(i.key()));
96 ++i;
97 }
98 directionEditor->setCurrentIndex(VProperty::d_ptr->VariantValue.toInt());
99 connect(directionEditor, static_cast<void (QComboBox::*)(int)>(&QComboBox::currentIndexChanged),
100 this, &DirectionProperty::currentIndexChanged);
101
102 VProperty::d_ptr->editor = directionEditor;
103 return VProperty::d_ptr->editor;
104}
