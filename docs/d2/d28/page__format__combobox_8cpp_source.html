183QString PageFormatCombobox::MakeHelpTemplateList()
184{
185 QString out = "\n";
186
187 auto cntr = static_cast<int>(PaperSizeFormat::A0);
188 foreach(auto& v, initFormats())
189 {
190 if (cntr <= static_cast<int>(PaperSizeFormat::Roll44in))// Don't include custom template
191 {
192 out += QLatin1String("\t") + v.first + QLatin1String(" = ") +
193 QString::number(static_cast<int>(v.second)) + QLatin1String("\n");
194 }
195 }
196 return out;
197}

199/**
200 * Called when the width has changed. This method sets the current width to the value chosen or even
201 * offers a dialog to the user that allows him/ her to choose an individual width.

203void PageFormatCombobox::updatePageFormat(int index)
204{
205 QVariant format = itemData(index);
206 if(format != QVariant::Invalid )
207 {
208 m_currentFormat = static_cast<PaperSizeFormat>(this->currentData().toInt());
209 }
210
211 emit pageFormatChanged(m_currentFormat);
212}
