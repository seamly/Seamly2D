36SeamlyWelcomeDialog::SeamlyWelcomeDialog(QWidget *parent)
37 : QDialog(parent)
38 , ui(new Ui::SeamlyWelcomeDialog)
39 , m_langChanged(false)
40 , m_selectionSoundChanged(false)
41 , settings(qApp->Seamly2DSettings())
42{
43 ui->setupUi(this);
44 setWindowFlags(windowFlags() & ~Qt::WindowContextHelpButtonHint);
45
46 //-------------------- Units setup
47 initUnits(MeasurementsType::Individual);
48
49 //-------------------- Decimal separator setup
50 ui->separator_CheckBox->setText(tr("User locale") + QString(" (%1)").arg(QLocale().decimalPoint()));
51 ui->separator_CheckBox->setChecked(settings->getOsSeparator());
52 connect(ui->separator_CheckBox, &QCheckBox::stateChanged, this, &SeamlyWelcomeDialog::seperatorChanged);
53
54 //-------------------- Languages setup

56 connect(ui->language_ComboBox, static_cast<void (QComboBox::*)(int)>(&QComboBox::currentIndexChanged),

59 m_langChanged = true;
60 });
61
62 //-------------------- Selection sound
63 int index = ui->selectionSound_ComboBox->findText(settings->getSound());
64 if (index != -1)
65 {
66 ui->selectionSound_ComboBox->setCurrentIndex(index);
67 }
68 connect(ui->selectionSound_ComboBox, static_cast<void (QComboBox::*)(int)>(&QComboBox::currentIndexChanged), this, [this]()
69 {
70 m_selectionSoundChanged = true;
71 QSound::play("qrc:/sounds/" + ui->selectionSound_ComboBox->currentText() + ".wav");

73
74 ui->doNotShow_CheckBox->setChecked(settings->getShowWelcome());
75
76 QPushButton *ok_Button = ui->buttonBox->button(QDialogButtonBox::Ok);
77 SCASSERT(ok_Button != nullptr)

88// @brief apply apply dialog changes
89void SeamlyWelcomeDialog::apply()
90{
91 settings->SetUnit(qvariant_cast<QString>(ui->units_ComboBox->currentData()));
92 settings->setOsSeparator(ui->separator_CheckBox->isChecked());
93 settings->getOsSeparator() ? setLocale(QLocale()) : setLocale(QLocale::c());
94 settings->setShowWelcome(ui->doNotShow_CheckBox->isChecked());
95
96 if (m_langChanged)
97 {
98 const QString locale = qvariant_cast<QString>(ui->language_ComboBox->currentData());
99 settings->setLocale(locale);
100 m_langChanged = false;
101 }
102
103 if (m_selectionSoundChanged)
104 {
105 const QString sound = qvariant_cast<QString>(ui->selectionSound_ComboBox->currentText());
106 settings->setSelectionSound(sound);
107 m_selectionSoundChanged = false;
108 }
109
110 done(QDialog::Accepted);
111}
