93inline void noisyFailureMsgHandler(QtMsgType type, const QMessageLogContext &context, const QString &msg)
94{
95 // Why on earth didn't Qt want to make failed signal/slot connections qWarning?
96 if ((type == QtDebugMsg) && msg.contains(QStringLiteral("::connect")))
97 {
98 type = QtWarningMsg;
99 }
100
101#if defined(V_NO_ASSERT)
102 // I have decided to hide this annoing message for release builds.
103 if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QSslSocket: cannot resolve")))
104 {
105 type = QtDebugMsg;
106 }
107
108 if ((type == QtWarningMsg) && msg.contains(QStringLiteral("setGeometry: Unable to set geometry")))
109 {
110 type = QtDebugMsg;
111 }
112#endif //defined(V_NO_ASSERT)
113
114#if defined(Q_OS_MAC)
115 // Hide Qt bug 'Assertion when reading an icns file'
116 // https://bugreports.qt.io/browse/QTBUG-45537
117 // Remove after Qt fix will be released
118 if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QICNSHandler::read()")))
119 {
120 type = QtDebugMsg;
121 }
122
123 // See issue #568
124 if (msg.contains(QStringLiteral("Error receiving trust for a CA certificate")))
125 {
126 type = QtDebugMsg;
127 }
128#endif
129
130 // This is another one that doesn't make sense as just a debug message. pretty serious
131 // sign of a problem
132 // http://www.developer.nokia.com/Community/Wiki/QPainter::begin:Paint_device_returned_engine_%3D%3D_0_(Known_Issue)
133 if ((type == QtDebugMsg) && msg.contains(QStringLiteral("QPainter::begin"))
134 && msg.contains(QStringLiteral("Paint device returned engine")))
135 {
136 type = QtWarningMsg;
137 }
138
139 // This qWarning about "Cowardly refusing to send clipboard message to hung application..."
140 // is something that can easily happen if you are debugging and the application is paused.
141 // As it is so common, not worth popping up a dialog.
142 if ((type == QtWarningMsg) && msg.contains(QStringLiteral("QClipboard::event"))
143 && msg.contains(QStringLiteral("Cowardly refusing")))
144 {
145 type = QtDebugMsg;
146 }
147
148 // Only the GUI thread should display message boxes. If you are
149 // writing a multithreaded application and the error happens on
150 // a non-GUI thread, you'll have to queue the message to the GUI
151 QCoreApplication *instance = QCoreApplication::instance();
152 const bool isGuiThread = instance && (QThread::currentThread() == instance->thread());
153
154 {
155 QString debugdate = "[" + QDateTime::currentDateTime().toString(QStringLiteral("yyyy.MM.dd hh:mm:ss"));
156
157 switch (type)
158 {
159 case QtDebugMsg:
160 debugdate += QString(":DEBUG:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
161 .arg(context.function).arg(context.category).arg(msg);
162 vStdOut() << QApplication::translate("vNoisyHandler", "DEBUG:") << msg << "\n";
163 break;
164 case QtWarningMsg:
165 debugdate += QString(":WARNING:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
166 .arg(context.function).arg(context.category).arg(msg);
167 vStdErr() << QApplication::translate("vNoisyHandler", "WARNING:") << msg << "\n";
168 break;
169 case QtCriticalMsg:
170 debugdate += QString(":CRITICAL:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
171 .arg(context.function).arg(context.category).arg(msg);
172 vStdErr() << QApplication::translate("vNoisyHandler", "CRITICAL:") << msg << "\n";
173 break;
174 case QtFatalMsg:
175 debugdate += QString(":FATAL:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
176 .arg(context.function).arg(context.category).arg(msg);
177 vStdErr() << QApplication::translate("vNoisyHandler", "FATAL:") << msg << "\n";
178 break;
179 #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
180 case QtInfoMsg:
181 debugdate += QString(":INFO:%1(%2)] %3: %4: %5").arg(context.file).arg(context.line)
182 .arg(context.function).arg(context.category).arg(msg);
183 vStdOut() << QApplication::translate("vNoisyHandler", "INFO:") << msg << "\n";
184 break;
185 #endif
186 default:
187 break;
188 }
189
190 (*qApp->logFile()) << debugdate << Qt::endl;
191 }
192
193 if (isGuiThread)
194 {
195 // fixme: trying to make sure that no save/load dialogs are opened, because an error message
196 // during them will lead to a crash
197 const bool topWinAllowsPop = (QApplication::activeModalWidget() == nullptr) ||
198 !QApplication::activeModalWidget()->inherits("QFileDialog");
199
200 QMessageBox messageBox;
201
202 switch (type)
203 {
204 case QtWarningMsg:
205 messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Warning"));
206 messageBox.setIcon(QMessageBox::Warning);
207 break;
208 case QtCriticalMsg:
209 messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Critical Error"));
210 messageBox.setIcon(QMessageBox::Critical);
211 break;
212 case QtFatalMsg:
213 messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Fatal Error"));
214 messageBox.setIcon(QMessageBox::Critical);
215 break;
216 #if QT_VERSION > QT_VERSION_CHECK(5, 4, 2)
217 case QtInfoMsg:
218 messageBox.setWindowTitle(QApplication::translate("vNoisyHandler", "Information"));
219 messageBox.setIcon(QMessageBox::Information);
220 break;
221 #endif
222 case QtDebugMsg:
223 default:
224 break;
225 }
226
227 if (type == QtWarningMsg || type == QtCriticalMsg || type == QtFatalMsg)
228 {
229 if (Application2D::isGUIMode())
230 {
231 if (topWinAllowsPop)
232 {
233 messageBox.setText(msg);
234 messageBox.setStandardButtons(QMessageBox::Ok);
235 messageBox.setWindowModality(Qt::ApplicationModal);
236 messageBox.setModal(true);
237 #ifndef QT_NO_CURSOR
238 QGuiApplication::setOverrideCursor(Qt::ArrowCursor);
239 #endif
240 messageBox.setWindowFlags(messageBox.windowFlags() & ~Qt::WindowContextHelpButtonHint);
241 messageBox.exec();
242 #ifndef QT_NO_CURSOR
243 QGuiApplication::restoreOverrideCursor();
244 #endif
245 }
246 }
247 }
248
249 if (QtFatalMsg == type)
250 {
251 abort();
252 }
253 }
254 else
255 {
256 if( QtDebugMsg != type && QtWarningMsg != type )
257 {
258 abort(); // be NOISY unless overridden!
259 }
260 }
261}

337// @brief notify Reimplemented from QApplication::notify().
338// @param receiver receiver.
339// @param event event.
340// @return value that is returned from the receiver's event handler.
341// reimplemented from QApplication so we can throw exceptions in slots
342bool Application2D::notify(QObject *receiver, QEvent *event)
343{
344 try
345 {
346 return QApplication::notify(receiver, event);
347 }
348 catch (const VExceptionObjectError &error)
349 {
350 qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error parsing file. Program will be terminated.")), //-V807
351 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
352 exit(V_EX_DATAERR);
353 }
354 catch (const VExceptionBadId &error)
355 {
356 qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error bad id. Program will be terminated.")),
357 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
358 exit(V_EX_DATAERR);
359 }
360 catch (const VExceptionConversionError &error)
361 {
362 qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error can't convert value. Program will be terminated.")),
363 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
364 exit(V_EX_DATAERR);
365 }
366 catch (const VExceptionEmptyParameter &error)
367 {
368 qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error empty parameter. Program will be terminated.")),
369 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
370 exit(V_EX_DATAERR);
371 }
372 catch (const VExceptionWrongId &error)
373 {
374 qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Error wrong id. Program will be terminated.")),
375 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
376 exit(V_EX_DATAERR);
377 }
378 catch (const VExceptionToolWasDeleted &error)
379 {
380 qCCritical(vApp, "%s\n\n%s\n\n%s",
381 qUtf8Printable("Unhadled deleting tool. Continue use object after deleting"),
382 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
383 exit(V_EX_DATAERR);
384 }
385 catch (const VException &error)
386 {
387 qCCritical(vApp, "%s\n\n%s\n\n%s", qUtf8Printable(tr("Something's wrong!!")),
388 qUtf8Printable(error.ErrorMessage()), qUtf8Printable(error.DetailedInformation()));
389 return true;
390 }
391 // These last two cases are special. I found that we can't show a modal dialog here with an error message.
392 // Somehow the program doesn't wait until an error dialog is closed, but if the exception is ignored
393 // the program will hang.
394 catch (const qmu::QmuParserError &error)
395 {
396 qCCritical(vApp, "%s", qUtf8Printable(tr("Parser error: %1. Program will be terminated.").arg(error.GetMsg())));
397 exit(V_EX_DATAERR);
398 }
399 catch (std::exception &error)
400 {
401 qCCritical(vApp, "%s", qUtf8Printable(tr("Exception thrown: %1. Program will be terminated.").arg(error.what())));
402 exit(V_EX_SOFTWARE);
403 }
404 return false;
405}

491//---------------------------------------------------------------------------------------------------------------------
492void Application2D::beginLogging()
493{
494 VlpCreateLock(m_lockLog, logPath(), [this](){return new QFile(logPath());});
495
496 if (m_lockLog->IsLocked())
497 {
498 if (m_lockLog->GetProtected()->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
499 {
500 m_out.reset(new QTextStream(m_lockLog->GetProtected().get()));
501 qInstallMessageHandler(noisyFailureMsgHandler);
502 qCInfo(vApp, "Log file %s was locked.", qUtf8Printable(logPath()));
503 }
504 else
505 {
506 qCWarning(vApp, "Error opening log file \'%s\'. All debug output redirected to console.",
507 qUtf8Printable(logPath()));
508 }
509 }
510 else
511 {
512 qCWarning(vApp, "Failed to lock %s", qUtf8Printable(logPath()));
513 }
514}

516//---------------------------------------------------------------------------------------------------------------------
517void Application2D::clearOldLogs() const
518{
519 QDir logsDir(logDirPath());
520 logsDir.setNameFilters(QStringList("*.log"));
521 logsDir.setCurrent(logDirPath());
522
523 const QStringList allFiles = logsDir.entryList(QDir::NoDotAndDotDot | QDir::Files);
524 if (allFiles.isEmpty() == false)
525 {
526 qCDebug(vApp, "Clearing old logs");
527 for (int i = 0, sz = allFiles.size(); i < sz; ++i)
528 {
529 auto fn = allFiles.at(i);
530 QFileInfo info(fn);
531 if (info.birthTime().daysTo(QDateTime::currentDateTime()) >= DAYS_TO_KEEP_LOGS)
532 {
533 VLockGuard<QFile> tmp(info.absoluteFilePath(), [&fn](){return new QFile(fn);});
534 if (tmp.GetProtected() != nullptr)
535 {
536 if (tmp.GetProtected()->remove())
537 {
538 qCDebug(vApp, "Deleted %s", qUtf8Printable(info.absoluteFilePath()));
539 }
540 else
541 {
542 qCWarning(vApp, "Could not delete %s", qUtf8Printable(info.absoluteFilePath()));
543 }
544 }
545 else
546 {
547 qCWarning(vApp, "Failed to lock %s", qUtf8Printable(info.absoluteFilePath()));
548 }
549 }
550 }
551 }
552 else
553 {
554 qCDebug(vApp, "There are no old logs.");
555 }
556}

643bool Application2D::event(QEvent *event)
644{
645 switch(event->type())
646 {
647 // In Mac OS X the QFileOpenEvent event is generated when user performs "Open With" from Finder (this event is
648 // Mac specific).
649 case QEvent::FileOpen:
650 {
651 QFileOpenEvent *fileOpenEvent = static_cast<QFileOpenEvent *>(event);
652 const QString macFileOpen = fileOpenEvent->file();
653 if(!macFileOpen.isEmpty())
654 {
655 MainWindow *window = qobject_cast<MainWindow*>(mainWindow);
656 if (window)
657 {
658 window->LoadPattern(macFileOpen); // open file in existing window
659 }
660 return true;
661 }
662 break;
663 }
664#if defined(Q_OS_MAC)
665 case QEvent::ApplicationActivate:
666 {
667 if (mainWindow && not mainWindow->isMinimized())
668 {
669 mainWindow->show();
670 }
671 return true;
672 }
673#endif //defined(Q_OS_MAC)
674 default:
675 return VAbstractApplication::event(event);
676 }
677 return VAbstractApplication::event(event);
678}
