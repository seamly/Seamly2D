157//---------------------------------------------------------------------------------------------------------------------
158void DialogPointFromArcAndTangent::ChosenObject(quint32 id, const SceneObject &type)
159{
160 if (prepare == false)// After first choose we ignore all objects
161 {
162 if (type == SceneObject::Point || type == SceneObject::Arc)
163 {
164 VisToolPointFromArcAndTangent *point = qobject_cast<VisToolPointFromArcAndTangent *>(vis);
165 SCASSERT(point != nullptr)
166
167 switch (number)
168 {
169 case 0:
170 if (type == SceneObject::Point)
171 {
172 if (SetObject(id, ui->comboBoxTangentPoint, tr("Select an arc")))
173 {
174 number++;
175 point->VisualMode(id);
176 }
177 }
178 break;
179 case 1:
180 if (type == SceneObject::Arc)
181 {
182 if (SetObject(id, ui->comboBoxArc, ""))
183 {
184 number = 0;
185 point->setArcId(id);
186 point->RefreshGeometry();
187 prepare = true;
188 DialogAccepted();
189 }
190 }
191 break;
192 default:
193 break;
194 }
195 }
196 }
197}
