454bool DialogTool::eventFilter(QObject *object, QEvent *event)
455{
456 if (QPlainTextEdit *plainTextEdit = qobject_cast<QPlainTextEdit *>(object))
457 {
458 if (event->type() == QEvent::KeyPress)
459 {
460 QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
461 if ((keyEvent->key() == Qt::Key_Enter) || (keyEvent->key() == Qt::Key_Return))
462 {
463 // Ignore Enter key
464 return true;
465 }
466 else if ((keyEvent->key() == Qt::Key_Period) && (keyEvent->modifiers() & Qt::KeypadModifier))
467 {
468 if (qApp->Settings()->getOsSeparator())
469 {
470 plainTextEdit->insertPlainText(QLocale().decimalPoint());
471 }
472 else
473 {
474 plainTextEdit->insertPlainText(QLocale::c().decimalPoint());
475 }
476 return true;
477 }
478 }
479 }
480 else
481 {
482 // pass the event on to the parent class
483 return QDialog::eventFilter(object, event);
484 }
485 return false;// pass the event to the widget
486}

502//---------------------------------------------------------------------------------------------------------------------
503int DialogTool::FindNotExcludedNodeDown(QListWidget *listWidget, int candidate)
504{
505 SCASSERT(listWidget != nullptr);
506
507 int index = -1;
508 if (candidate < 0 || candidate >= listWidget->count())
509 {
510 return index;
511 }
512
513 int i = candidate;
514 VPieceNode rowNode;
515 do
516 {
517 const QListWidgetItem *rowItem = listWidget->item(i);
518 SCASSERT(rowItem != nullptr);
519 rowNode = qvariant_cast<VPieceNode>(rowItem->data(Qt::UserRole));
520
521 if (!rowNode.isExcluded())
522 {
523 index = i;
524 }
525
526 ++i;
527 }
528 while (rowNode.isExcluded() && i < listWidget->count());
529
530 return index;
531}

533//---------------------------------------------------------------------------------------------------------------------
534int DialogTool::FindNotExcludedNodeUp(QListWidget *listWidget, int candidate)
535{
536 SCASSERT(listWidget != nullptr);
537
538 int index = -1;
539 if (candidate < 0 || candidate >= listWidget->count())
540 {
541 return index;
542 }
543
544 int i = candidate;
545 VPieceNode rowNode;
546 do
547 {
548 const QListWidgetItem *rowItem = listWidget->item(i);
549 SCASSERT(rowItem != nullptr);
550 rowNode = qvariant_cast<VPieceNode>(rowItem->data(Qt::UserRole));
551
552 if (!rowNode.isExcluded())
553 {
554 index = i;
555 }
556
557 --i;
558 }
559 while (rowNode.isExcluded() && i > -1);
560
561 return index;
562}
