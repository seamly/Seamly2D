152/**
153 * @brief Create help create tool.
154 * @param _id tool id, 0 if tool doesn't exist yet.
155 * @param pointName point name.
156 * @param formula string with formula length first splinePath.
157 * @param splinePathId id of splinePath in data container.
158 * @param mx label bias x axis.
159 * @param my label bias y axis.
160 * @param scene pointer to scene.
161 * @param doc dom document container.
162 * @param data container with variables.
163 * @param parse parser file mode.
164 * @param typeCreation way we create this tool.
165 */
166VToolCutSplinePath* VToolCutSplinePath::Create(const quint32 _id, const QString &pointName, QString &direction,
167 QString &formula, quint32 splinePathId, qreal mx, qreal my,
168 bool showPointName, VMainGraphicsScene *scene,
169 VAbstractPattern *doc, VContainer *data, const Document &parse,
170 const Source &typeCreation)
171{
172 const auto splPath = data->GeometricObject<VAbstractCubicBezierPath>(splinePathId);
173 SCASSERT(splPath != nullptr)
174
175 qreal splLength = qApp->fromPixel(splPath->GetLength());
176 QString adjFormula = formula;
177 if (direction == "backward")
178 {
179 adjFormula = QString("%1 - %2").arg(splLength).arg(formula);
180 }
181
182 const qreal length = CheckFormula(_id, adjFormula, data);
183
184 quint32 id = _id;
185 VSplinePath *splPath1 = nullptr;
186 VSplinePath *splPath2 = nullptr;
187
188 VPointF *p = VToolCutSplinePath::CutSplinePath(qApp->toPixel(length), splPath, pointName, &splPath1, &splPath2);
189 p->setShowPointName(showPointName);
190
191 SCASSERT(splPath1 != nullptr)
192 SCASSERT(splPath2 != nullptr)
193 SCASSERT(p != nullptr)
194
195 p->setMx(mx);
196 p->setMy(my);
197
198 if (typeCreation == Source::FromGui)
199 {
200 id = data->AddGObject(p);
201
202 data->AddSpline(QSharedPointer<VAbstractBezier>(splPath1), NULL_ID, id);
203 data->AddSpline(QSharedPointer<VAbstractBezier>(splPath2), NULL_ID, id);
204 }
205 else
206 {
207 data->UpdateGObject(id, p);
208
209 data->AddSpline(QSharedPointer<VAbstractBezier>(splPath1), NULL_ID, id);
210 data->AddSpline(QSharedPointer<VAbstractBezier>(splPath2), NULL_ID, id);
211
212 if (parse != Document::FullParse)
213 {
214 doc->UpdateToolData(id, data);
215 }
216 }
217
218 if (parse == Document::FullParse)
219 {
220 VDrawTool::AddRecord(id, Tool::CutSplinePath, doc);
221 VToolCutSplinePath *point = new VToolCutSplinePath(doc, data, id, direction, formula, splinePathId, typeCreation);
222 scene->addItem(point);
223 InitToolConnections(scene, point);
224 VAbstractPattern::AddTool(id, point);
225 doc->IncrementReferens(splPath->getIdTool());
226 return point;
227 }
228 return nullptr;
229}

238VPointF *VToolCutSplinePath::CutSplinePath(qreal length, const QSharedPointer<VAbstractCubicBezierPath> &splPath,
239 const QString &pName, VSplinePath **splPath1, VSplinePath **splPath2)
240{
241 SCASSERT(splPath != nullptr)
242
243 QPointF spl1p2, spl1p3, spl2p2, spl2p3;
244 qint32 p1 = 0, p2 = 0;
245
246 const QPointF point = splPath->CutSplinePath(length, p1, p2, spl1p2, spl1p3, spl2p2, spl2p3);
247 VPointF *p = new VPointF(point);
248 p->setName(pName);
249
250 const QVector<VSplinePoint> points = splPath->GetSplinePath();
251
252 const VSplinePoint splP1 = points.at(p1);
253 const VSplinePoint splP2 = points.at(p2);
254 const VSpline spl1 = VSpline(splP1.P(), spl1p2, spl1p3, *p);
255 const VSpline spl2 = VSpline(*p, spl2p2, spl2p3, splP2.P());
256
257 *splPath1 = new VSplinePath();
258 *splPath2 = new VSplinePath();
259
260 for (qint32 i = 0; i < points.size(); i++)
261 {
262 if (i <= p1 && i < p2)
263 {
264 if (i == p1)
265 {
266 const qreal angle1 = spl1.GetStartAngle()+180;
267 const QString angle1F = QString().number(angle1);
268
269 (*splPath1)->append(VSplinePoint(splP1.P(), angle1, angle1F, spl1.GetStartAngle(),
270 spl1.GetStartAngleFormula(), splP1.Length1(), splP1.Length1Formula(),
271 spl1.GetC1Length(), spl1.GetC1LengthFormula()));
272
273 const qreal angle2 = spl1.GetEndAngle()+180;
274 const QString angle2F = QString().number(angle2);
275
276 const auto cutPoint = VSplinePoint(*p, spl1.GetEndAngle(), spl1.GetEndAngleFormula(), angle2, angle2F,
277 spl1.GetC2Length(), spl1.GetC2LengthFormula(), spl2.GetC1Length(),
278 spl2.GetC1LengthFormula());
279 (*splPath1)->append(cutPoint);
280 continue;
281 }
282 (*splPath1)->append(points.at(i));
283 }
284 else
285 {
286 if (i == p2)
287 {
288 const qreal angle1 = spl2.GetStartAngle()+180;
289 const QString angle1F = QString().number(angle1);
290
291 const auto cutPoint = VSplinePoint(*p, angle1, angle1F, spl2.GetStartAngle(),
292 spl2.GetStartAngleFormula(), spl1.GetC2Length(),
293 spl1.GetC2LengthFormula(), spl2.GetC1Length(),
294 spl2.GetC1LengthFormula());
295
296 (*splPath2)->append(cutPoint);
297
298 const qreal angle2 = spl2.GetEndAngle()+180;
299 const QString angle2F = QString().number(angle2);
300
301 (*splPath2)->append(VSplinePoint(splP2.P(), spl2.GetEndAngle(), spl2.GetEndAngleFormula(), angle2,
302 angle2F, spl2.GetC2Length(), spl2.GetC2LengthFormula(),
303 splP2.Length2(), splP2.Length2Formula()));
304 continue;
305 }
306 (*splPath2)->append(points.at(i));
307 }
308 }
309
310 return p;
311}

385QString VToolCutSplinePath::makeToolTip() const
386{
387 const auto splPath = VAbstractTool::data.GeometricObject<VAbstractCubicBezierPath>(curveCutId);
388
389 const QString expression = qApp->translateVariables()->FormulaToUser(formula, qApp->Settings()->getOsSeparator());
390 const qreal length = Visualization::FindVal(expression, VAbstractTool::data.DataVariables());
391
392 VSplinePath *splPath1 = nullptr;
393 VSplinePath *splPath2 = nullptr;
394 VPointF *p = VToolCutSplinePath::CutSplinePath(qApp->toPixel(length), splPath, "X", &splPath1, &splPath2);
395 delete p; // Don't need this point
396
397 const QString curveStr = tr("Curve");
398 const QString lengthStr = tr("length");
399
400 const QString toolTip = QString("<table style=font-size:11pt; font-weight:600>"
401 "<tr> <td><b>%6:</b> %7</td> </tr>"
402 "<tr> <td><b>%1:</b> %2 %3</td> </tr>"
403 "<tr> <td><b>%8:</b> %9</td> </tr>"
404 "<tr> <td><b>%4:</b> %5 %3</td> </tr>"
405 "</table>")
406 .arg(curveStr + QLatin1String("1 ") + lengthStr)
407 .arg(qApp->fromPixel(splPath1->GetLength()))
408 .arg(UnitsToStr(qApp->patternUnit(), true))
409 .arg(curveStr + QLatin1String("2 ") + lengthStr)
410 .arg(qApp->fromPixel(splPath2->GetLength()))
411 .arg(curveStr + QLatin1String(" 1") + tr("label"))
412 .arg(splPath1->name())
413 .arg(curveStr + QLatin1String(" 2") + tr("label"))
414 .arg(splPath2->name());
415
416 delete splPath1;
417 delete splPath2;
418
419 return toolTip;
