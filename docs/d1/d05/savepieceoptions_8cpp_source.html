87void SavePieceOptions::undo()
88{
89 qCDebug(vUndo, "Undo.");
90
91 QDomElement domElement = doc->elementById(nodeId, VAbstractPattern::TagPiece);
92 if (domElement.isElement())
93 {
94 PatternPieceTool::AddAttributes(doc, domElement, nodeId, m_oldPiece);
95 doc->RemoveAllChildren(domElement);//Very important to clear before rewrite
96 PatternPieceTool::AddPatternPieceData(doc, domElement, m_oldPiece);
97 PatternPieceTool::AddPatternInfo(doc, domElement, m_oldPiece);
98 PatternPieceTool::AddGrainline(doc, domElement, m_oldPiece);
99 PatternPieceTool::AddNodes(doc, domElement, m_oldPiece);
100 PatternPieceTool::AddCSARecords(doc, domElement, m_oldPiece.GetCustomSARecords());
101 PatternPieceTool::AddInternalPaths(doc, domElement, m_oldPiece.GetInternalPaths());
102 PatternPieceTool::addAnchors(doc, domElement, m_oldPiece.getAnchors());
103
104 IncrementReferences(m_oldPiece.MissingNodes(m_newPiece));
105 IncrementReferences(m_oldPiece.MissingCSAPath(m_newPiece));
106 IncrementReferences(m_oldPiece.MissingInternalPaths(m_newPiece));
107 IncrementReferences(m_oldPiece.missingAnchors(m_newPiece));
108 emit NeedLiteParsing(Document::LiteParse);
109 }
110 else
111 {
112 qCWarning(vUndo, "Can't find piece with id = %u.", nodeId);
113 return;
114 }
115}

150bool SavePieceOptions::mergeWith(const QUndoCommand *command)
151{
152 const SavePieceOptions *saveCommand = static_cast<const SavePieceOptions *>(command);
153 SCASSERT(saveCommand != nullptr);
154 const quint32 id = saveCommand->pieceId();
155
156 if (id != nodeId)
157 {
158 return false;
159 }
160
161 m_newPiece = saveCommand->getNewPiece();
162 return true;
163}
