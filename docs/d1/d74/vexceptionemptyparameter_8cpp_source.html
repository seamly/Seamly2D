67VExceptionEmptyParameter::VExceptionEmptyParameter(const QString &what, const QString &name,
68 const QDomElement &domElement)
69 : VException(what)
70 , name(name)
71 , tagText(QString())
72 , tagName(QString())
73 , lineNumber(-1)
74{
75 Q_ASSERT_X(not domElement.isNull(), Q_FUNC_INFO, "domElement is null");
76 Q_ASSERT_X(not name.isEmpty(), Q_FUNC_INFO, "Parameter name is empty");

81}
