116// @brief Paint paints the spline control handles. Paints the line from the spline to the control point first,
117// then calls the SceneRect widget paint to paint the control point.
118// @param painter Qpainter.
119// @param option QStyleOptionGraphicsItem.
120// @param widget parent QWidget.
121void VControlPointSpline::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
122{
123 QPen lPen = m_controlLine->pen();
124 lPen.setColor(correctColor(m_controlLine, Qt::black));
125
126 QPointF p1, p2;
127 VGObject::LineIntersectCircle(QPointF(), scaledRadius(sceneScale(scene())),
128 QLineF(QPointF(), m_controlLine->line().p1()), p1, p2);
129 QLineF const line(m_controlLine->line().p1(), p1);
130
131 m_controlLine->setPen(lPen);
132 m_controlLine->setLine(line);
133
134 SceneRect::paint(painter, option, widget);
135}

159// @brief itemChange handle item change.
160// @param change change.
161// @param value value.
162// @return value.
163QVariant VControlPointSpline::itemChange(QGraphicsItem::GraphicsItemChange change, const QVariant &value)
164{
165 switch (change)
166 {
167 case ItemPositionChange:
168 {
169 if (!m_freeAngle || !m_freeLength)
170 {
171 const QPointF splPoint = m_controlLine->line().p1() + pos();
172
173 QLineF newLine(splPoint, value.toPointF()); // value - new position.
174 QLineF oldLine(splPoint, pos()); // pos() - old position.
175
176 if (!m_freeAngle)
177 {
178 newLine.setAngle(oldLine.angle());
179 }
180
181 if (!m_freeLength)
182 {
183 newLine.setLength(oldLine.length());
184 }
185
186 return newLine.p2();
187 }
188
189 break;
190 }
191 case ItemPositionHasChanged:
192 {
193 // Each time we move something we call recalculation scene rect. In some cases this can cause moving
194 // objects positions. And this cause infinite redrawing. That's why we wait the finish of saving the last
195 // move.
196 static bool changeFinished = true;
197 if (changeFinished)
198 {
199 changeFinished = false;
200 // value - new position.
201 emit controlPointPositionChanged(m_splineIndex, m_position, value.toPointF());
202 if (scene())
203 {
204 const QList<QGraphicsView *> viewList = scene()->views();
205 if (!viewList.isEmpty())
206 {
207 if (QGraphicsView *view = viewList.at(0))
208 {
209 const int xmargin = 50;
210 const int ymargin = 50;
211
212 const QRectF viewRect = VMainGraphicsView::SceneVisibleArea(view);
213 const QRectF itemRect = mapToScene(boundingRect()).boundingRect();
214
215 // If item's rect is bigger than view's rect ensureVisible works very unstable.
216 if (itemRect.height() + 2*ymargin < viewRect.height() &&
217 itemRect.width() + 2*xmargin < viewRect.width())
218 {
219 view->ensureVisible(itemRect, xmargin, ymargin);
220 }
221 else
222 {
223 // Ensure visible only small rect around a cursor
224 VMainGraphicsScene *currentScene = qobject_cast<VMainGraphicsScene *>(scene());
225 SCASSERT(currentScene)
226 const QPointF cursorPosition = currentScene->getScenePos();
227 view->ensureVisible(QRectF(cursorPosition.x()-5, cursorPosition.y()-5, 10, 10));
228 }
229 }
230 }
231 }
232 changeFinished = true;
233 }
234 break;
235 }
236 case QGraphicsItem::ItemSelectedHasChanged:
237 emit selected(value.toBool());
238 break;
239 default:
240 break;
241 }
242 return SceneRect::itemChange(change, value);
243}

298// @brief setCtrlLine set the geometry of the control handle line based on the control and spline points.
299// @param controlPoint end of control line.
300// @param splinePoint start point of the control line.
301void VControlPointSpline::setCtrlLine(const QPointF &controlPoint, const QPointF &splinePoint)
302{
303 QPointF p1, p2;
304 VGObject::LineIntersectCircle(QPointF(), scaledRadius(sceneScale(scene())),
305 QLineF(QPointF(), splinePoint - controlPoint), p1, p2);
306
307 QLineF const line(splinePoint - controlPoint, p1);
308 m_controlLine->setFlag(QGraphicsItem::ItemSendsGeometryChanges, false); //prevent paint recursion
309 m_controlLine->setLine(QLineF(splinePoint - controlPoint, p1));
310 m_controlLine->setVisible(!line.isNull());
311 m_controlLine->setFlag(QGraphicsItem::ItemSendsGeometryChanges, true);
312}

314// @brief refreshCtrlPoint refresh the control point.
315// @param splineIndex index spline in list.
316// @param pos position point in spline.
317// @param controlPoint control point.
318// @param splinePoint spline point.
319void VControlPointSpline::refreshCtrlPoint(const qint32 &splineIndex, SplinePointPosition pos,
320 const QPointF &controlPoint, const QPointF &splinePoint, bool freeAngle,
321 bool freeLength)
322{
323 if (this->m_splineIndex == splineIndex && this->m_position == pos)
324 {
325 this->m_freeAngle = freeAngle;
326 this->m_freeLength = freeLength;
327 this->setPos(controlPoint);
328 setCtrlLine(controlPoint, splinePoint);
329 }
330}
