526 *
527 * Because translated string can have different length compare to original need make correction after each translation.
528 * If bias = 0 correction will not happens.
529 *
530 * @param position position currecnt token in expression
531 * @param bias difference between original token length and translated
532 * @param tokens all tokens
533 * @param numbers all numbers
534 */
535void VTranslateVars::CorrectionsPositions(int position, int bias, QMap<int, QString> &tokens,
536 QMap<int, QString> &numbers) const
537{
538 if (bias == 0)
539 {
540 return;// Nothing to correct;
541 }
542
543 BiasTokens(position, bias, tokens);
544 BiasTokens(position, bias, numbers);
545}

548/**
549 * @brief BiasTokens change position for each token that have position more then "position".
550 * @param position token position
551 * @param bias difference between original token length and translated
552 * @param tokens all tokens
553 */
554void VTranslateVars::BiasTokens(int position, int bias, QMap<int, QString> &tokens)
555{
556 QMap<int, QString> newTokens;
557 QMap<int, QString>::const_iterator i = tokens.constBegin();
558 while (i != tokens.constEnd())
559 {
560 if (i.key()<= position)
561 { // Tokens before position "position" did not change his positions.
562 newTokens.insert(i.key(), i.value());
563 }
564 else
565 {
566 newTokens.insert(i.key()-bias, i.value());
567 }
568 ++i;
569 }
570 tokens = newTokens;
571}

576/**
577 * @brief VariablesFromUser translate variable to internal look.
578 * @param newFormula [in|out] expression to translate
579 * @param position token position
580 * @param token token to translate
581 * @param bias hold change of length between translated and origin token string
582 * @return true if was found variable with same name.
583 */
584bool VTranslateVars::VariablesFromUser(QString &newFormula, int position, const QString &token, int &bias) const
585{
586 const QString currentLengthTr = variables.value(currentLength).translate();
587 const QString currentSeamAllowanceTr = variables.value(currentSeamAllowance).translate();
588
589 QMap<QString, qmu::QmuTranslation>::const_iterator i = variables.constBegin();
590 while (i != variables.constEnd())
591 {
592 const qmu::QmuTranslation &var = i.value();
593 const QString varTr = var.translate();
594
595 if (token.indexOf(varTr) == 0)
596 {
597 if ((varTr == currentLengthTr || varTr == currentSeamAllowanceTr) && token != varTr)
598 {
599 ++i;
600 continue;
601 }
602
603 newFormula.replace(position, varTr.length(), i.key());
604 QString newToken = token;
605 newToken.replace(0, varTr.length(), i.key());
606 bias = token.length() - newToken.length();
607 return true;
608 }
609 ++i;
610 }
611 return false;
612}

836/**
837 * @brief FormulaFromUser replace all known tokens in formula to internal look. Also change decimal
838 * separator in numbers.
839 * @param formula expression that need translate
840 * @throw qmu::QmuParserError in case of a wrong expression
841 * @return translated expression
842 */
843QString VTranslateVars::FormulaFromUser(const QString &formula, bool osSeparator) const
844{
845 if (formula.isEmpty())
846 {
847 return formula;
848 }
849 QString newFormula = formula;// Local copy for making changes
850
851 QScopedPointer<qmu::QmuTokenParser> cal(new qmu::QmuTokenParser(formula, osSeparator));// Eval formula
852 QMap<int, QString> tokens = cal->GetTokens();// Tokens (variables, measurements)
853 QMap<int, QString> numbers = cal->GetNumbers();// All numbers in expression for changing decimal separator
854 delete cal.take();
855
856 QList<int> tKeys = tokens.keys();// Take all tokens positions
857 QList<QString> tValues = tokens.values();
858 for (int i = 0; i < tKeys.size(); ++i)
859 {
860 int bias = 0;
861 if (MeasurementsFromUser(newFormula, tKeys.at(i), tValues.at(i), bias))
862 {
863 if (bias != 0)
864 {// Translated token has different length than original. Position next tokens need to be corrected.
865 CorrectionsPositions(tKeys.at(i), bias, tokens, numbers);
866 tKeys = tokens.keys();
867 tValues = tokens.values();
868 }
869 continue;
870 }
871
872 if (VariablesFromUser(newFormula, tKeys.at(i), tValues.at(i), bias))
873 {
874 if (bias != 0)
875 {// Translated token has different length than original. Position next tokens need to be corrected.
876 CorrectionsPositions(tKeys.at(i), bias, tokens, numbers);
877 tKeys = tokens.keys();
878 tValues = tokens.values();
879 }
880 continue;
881 }
882
883 if (PostfixOperatorsFromUser(newFormula, tKeys.at(i), tValues.at(i), bias))
884 {
885 if (bias != 0)
886 {// Translated token has different length than original. Position next tokens need to be corrected.
887 CorrectionsPositions(tKeys.at(i), bias, tokens, numbers);
888 tKeys = tokens.keys();
889 tValues = tokens.values();
890 }
891 continue;
892 }
893
894 if (FunctionsFromUser(newFormula, tKeys.at(i), tValues.at(i), bias))
895 {
896 if (bias != 0)
897 {// Translated token has different length than original. Position next tokens need to be corrected.
898 CorrectionsPositions(tKeys.at(i), bias, tokens, numbers);
899 tKeys = tokens.keys();
900 tValues = tokens.values();
901 }
902 continue;
903 }
904
905 if (tValues.at(i) == QLocale().negativeSign())
906 {// unary minus
907 newFormula.replace(tKeys.at(i), 1, '-');
908 }
909 }
910
911 QLocale loc = QLocale(); // User locale
912 if (loc != QLocale::c() && osSeparator)
913 {// User want use Os separator
914 QList<int> nKeys = numbers.keys();// Positions for all numbers in expression
915 QList<QString> nValues = numbers.values();
916 for (int i = 0; i < nKeys.size(); ++i)
917 {
918 loc = QLocale();// From system locale
919 bool ok = false;
920 const qreal d = loc.toDouble(nValues.at(i), &ok);
921 if (ok == false)
922 {
923 qDebug() << "Can't convert to double token"<<nValues.at(i);
924 continue;//Leave with out translation
925 }
926
927 loc = QLocale::c();// To internal locale
928 const QString dStr = loc.toString(d);// Internal look for number
929 newFormula.replace(nKeys.at(i), nValues.at(i).length(), dStr);
930 const int bias = nValues.at(i).length() - dStr.length();
931 if (bias != 0)
932 {// Translated number has different length than original. Position next tokens need to be corrected.
933 CorrectionsPositions(nKeys.at(i), bias, tokens, numbers);
934 nKeys = numbers.keys();
935 nValues = numbers.values();
936 }
937 }
938 }
939
940 return newFormula;
941}
