278//---------------------------------------------------------------------------------------------------------------------
279void DialogLineIntersectAxis::ShowDialog(bool click)
280{
281 if (prepare)
282 {
283 if (click)
284 {
285 // The check need to ignore first release of mouse button.
286 // User can select point by clicking on a label.
287 if (not m_firstRelease)
288 {
289 m_firstRelease = true;
290 return;
291 }
292
293 /*We will ignore click if poinet is in point circle*/
294 VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
295 SCASSERT(scene != nullptr)
296 const QSharedPointer<VPointF> point = data->GeometricObject<VPointF>(GetBasePointId());
297 QLineF line = QLineF(static_cast<QPointF>(*point), scene->getScenePos());
298
299 //Radius of point circle, but little bigger. Need handle with hover sizes.
300 if (line.length() <= defPointRadiusPixel*1.5)
301 {
302 return;
303 }
304 }
305
306 VisToolLineIntersectAxis *line = qobject_cast<VisToolLineIntersectAxis *>(vis);
307 SCASSERT(line != nullptr)
308
309 this->SetAngle(line->Angle());//Show in dialog angle what user choose
310 emit ToolTip("");
311
312 DialogAccepted();// Just set default values and don't show dialog
313 }
314}

316//---------------------------------------------------------------------------------------------------------------------
317void DialogLineIntersectAxis::ChosenObject(quint32 id, const SceneObject &type)
318{
319 if (prepare == false)// After first choose we ignore all objects
320 {
321 if (type == SceneObject::Point)
322 {
323 VisToolLineIntersectAxis *line = qobject_cast<VisToolLineIntersectAxis *>(vis);
324 SCASSERT(line != nullptr)
325
326 switch (number)
327 {
328 case (0):
329 if (SetObject(id, ui->comboBoxFirstLinePoint, tr("Select second point of line")))
330 {
331 number++;
332 line->VisualMode(id);
333 VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
334 SCASSERT(window != nullptr)
335 connect(line, &VisToolLineIntersectAxis::ToolTip, window, &VAbstractMainWindow::ShowToolTip);
336 }
337 break;
338 case (1):
339 if (getCurrentObjectId(ui->comboBoxFirstLinePoint) != id)
340 {
341 if (SetObject(id, ui->comboBoxSecondLinePoint, tr("Select axis point")))
342 {
343 number++;
344 line->setPoint2Id(id);
345 line->RefreshGeometry();
346 }
347 }
348 break;
349 case (2):
350 {
351 QSet<quint32> set;
352 set.insert(getCurrentObjectId(ui->comboBoxFirstLinePoint));
353 set.insert(getCurrentObjectId(ui->comboBoxSecondLinePoint));
354 set.insert(id);
355
356 if (set.size() == 3)
357 {
358 if (SetObject(id, ui->comboBoxAxisPoint, ""))
359 {
360 line->setAxisPointId(id);
361 line->RefreshGeometry();
362 prepare = true;
363 }
364 }
365 }
366 break;
367 default:
368 break;
369 }
370 }
371 }
372}
