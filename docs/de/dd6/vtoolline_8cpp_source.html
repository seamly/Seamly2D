159//---------------------------------------------------------------------------------------------------------------------
160/**
161 * @brief Create help create tool.
162 * @param _id tool id, 0 if tool doesn't exist yet.
163 * @param firstPoint id first line point.
164 * @param secondPoint id second line point.
165 * @param lineType line type.
166 * @param lineWeight line weight.
167 * @param lineColor line color.
168 * @param scene pointer to scene.
169 * @param doc dom document container.
170 * @param data container with variables.
171 * @param parse parser file mode.
172 * @param typeCreation way we create this tool.
173 */
174VToolLine * VToolLine::Create(const quint32 &_id, const quint32 &firstPoint, const quint32 &secondPoint,
175 const QString &lineType, const QString &lineWeight, const QString &lineColor,
176 VMainGraphicsScene *scene, VAbstractPattern *doc, VContainer *data,
177 const Document &parse, const Source &typeCreation)
178{
179 SCASSERT(scene != nullptr)
180 SCASSERT(doc != nullptr)
181 SCASSERT(data != nullptr)
182 quint32 id = _id;
183 if (typeCreation == Source::FromGui)
184 {
185 id = VContainer::getNextId();
186 data->AddLine(firstPoint, secondPoint);
187 }
188 else
189 {
190 VContainer::UpdateId(id);
191 data->AddLine(firstPoint, secondPoint);
192 if (parse != Document::FullParse)
193 {
194 doc->UpdateToolData(id, data);
195 }
196 }
197
198 if (parse == Document::FullParse)
199 {
200 VDrawTool::AddRecord(id, Tool::Line, doc);
201 VToolLine *line = new VToolLine(doc, data, id, firstPoint, secondPoint, lineType, lineWeight, lineColor, typeCreation);
202 scene->addItem(line);
203 InitDrawToolConnections(scene, line);
204 connect(scene, &VMainGraphicsScene::EnableLineItemSelection, line, &VToolLine::AllowSelecting);
205 connect(scene, &VMainGraphicsScene::EnableLineItemHover, line, &VToolLine::AllowHover);
206 VAbstractPattern::AddTool(id, line);
207
208 const QSharedPointer<VPointF> first = data->GeometricObject<VPointF>(firstPoint);
209 const QSharedPointer<VPointF> second = data->GeometricObject<VPointF>(secondPoint);
210
211 doc->IncrementReferens(first->getIdTool());
212 doc->IncrementReferens(second->getIdTool());
213 return line;
214 }
215 return nullptr;
216}

469QString VToolLine::makeToolTip() const
470{
471 const QSharedPointer<VPointF> first = VAbstractTool::data.GeometricObject<VPointF>(firstPoint);
472 const QSharedPointer<VPointF> second = VAbstractTool::data.GeometricObject<VPointF>(secondPoint);
473
474 const QLineF line(static_cast<QPointF>(*first), static_cast<QPointF>(*second));
475
476 const QString toolTip = QString("<table style=font-size:11pt; font-weight:600>"
477 "<tr> <td><b>%1:</b> %2 %3</td> </tr>"
478 "<tr> <td><b>%4:</b> %5°</td> </tr>"
479 "</table>")
480 .arg(tr("Length"))
481 .arg(qApp->fromPixel(line.length()))
482 .arg(UnitsToStr(qApp->patternUnit(), true))
483 .arg(tr("Angle"))
484 .arg(line.angle());
485 return toolTip;
486}
